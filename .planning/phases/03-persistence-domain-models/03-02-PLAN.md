import 'dart:math';
import 'package:mirkfall/infrastructure/ids/ulid.dart';
import 'package:test/test.dart';

void main() {
  group('Ulid.generate', () {
    test('produces exactly 26 chars', () {
      final id = Ulid.generate(now: DateTime(2026, 4, 18), rng: Random(1));
      expect(id.length, 26);
    });

    test('uses only Crockford base32 alphabet', () {
      const alphabet = '0123456789ABCDEFGHJKMNPQRSTVWXYZ';
      final id = Ulid.generate(now: DateTime(2026, 4, 18), rng: Random(1));
      for (final char in id.split('')) {
        expect(alphabet.contains(char), isTrue, reason: 'char "$char" not in alphabet');
      }
      // Forbidden: I, L, O, U
      for (final forbidden in ['I', 'L', 'O', 'U']) {
        expect(id.contains(forbidden), isFalse);
      }
    });

    test('k-sortable: later timestamp → lexically later string (with fixed rng)', () {
      final rng1 = Random(42);
      final rng2 = Random(42);
      final earlier = Ulid.generate(now: DateTime(2026, 4, 18), rng: rng1);
      final later = Ulid.generate(now: DateTime(2026, 5, 18), rng: rng2);
      // Time part = first 10 chars; random tail identical due to same seed.
      expect(earlier.compareTo(later), lessThan(0));
      expect(earlier.substring(10), later.substring(10),
          reason: 'random tail should be identical with fixed seed');
    });

    test('same seed + same timestamp → identical IDs (reproducibility for tests)', () {
      final id1 = Ulid.generate(now: DateTime(2026, 4, 18), rng: Random(7));
      final id2 = Ulid.generate(now: DateTime(2026, 4, 18), rng: Random(7));
      expect(id1, id2);
    });
  });
}

// random_id_generator.dart
class RandomIdGenerator implements IdGenerator {
  RandomIdGenerator([Random? rng]) : _rng = rng ?? Random.secure();
  final Random _rng;

  @override
  String newId(String prefix) =>
      '$prefix${Ulid.generate(now: DateTime.now().toUtc(), rng: _rng)}';
}

// seeded_id_generator.dart
class SeededIdGenerator implements IdGenerator {
  SeededIdGenerator({required int seed, DateTime? fixedNow})
      : _rng = Random(seed),
        _fixedNow = fixedNow;

  final Random _rng;
  final DateTime? _fixedNow;

  @override
  String newId(String prefix) {
    final now = _fixedNow ?? DateTime.now().toUtc();
    return '$prefix${Ulid.generate(now: now, rng: _rng)}';
  }
}
```

Rationale for `fixedNow` optional arg on SeededIdGenerator: makes tests even more deterministic (timestamp + random both fixed). Default null = wall-clock so the generator is still useful for end-to-end test harnesses that just need prefix safety.

/// Type-safe wrapper for session identifiers.
///
/// Zero-cost at runtime (Dart 3 extension type) — identical to `String` but
/// compile-time rejects cross-type assignment (`MarkerId` → `SessionId`).
extension type const SessionId(String value) {
  /// Prefix embedded in every session ID. Stored with the value (not only
  /// injected at JSON serialization) so a copy-pasted ID is immediately
  /// identifiable in logs, SQL inspector, and bug reports.
  static const String prefix = 'sess_';

  /// Returns true iff [value] starts with [prefix] and has the ULID suffix
  /// length (26 chars).
  bool get isValid =>
      value.startsWith(prefix) &amp;&amp; value.length == prefix.length + 26;
}
```

import 'dart:math';

/// (x, y, zoom) triple for a slippy-map tile.
class TilePosition {
  const TilePosition({required this.x, required this.y, required this.zoom});
  final int x;
  final int y;
  final int zoom;

  @override
  bool operator ==(Object other) =&gt;
      other is TilePosition &amp;&amp;
      other.x == x &amp;&amp; other.y == y &amp;&amp; other.zoom == zoom;

  @override
  int get hashCode =&gt; Object.hash(x, y, zoom);

  @override
  String toString() =&gt; 'TilePosition($x, $y, z=$zoom)';
}

/// Slippy-map tile math — OSM Web Mercator projection (EPSG:3857).
///
/// Reference: https://wiki.openstreetmap.org/wiki/Slippy_map_tilenames
class TileMath {
  TileMath._();

  /// Web-Mercator latitude clamp per OSM — poles undefined beyond ±85.0511°.
  static const double maxLatMercator = 85.05112878;

  /// Converts (lat, lon) to a tile position at [zoom]. Clamps latitude to
  /// Web-Mercator range so polar inputs do not produce NaN.
  static TilePosition latLonToTile({
    required double lat,
    required double lon,
    required int zoom,
  }) {
    final clampedLat = lat.clamp(-maxLatMercator, maxLatMercator);
    final n = pow(2.0, zoom).toDouble();
    final latRad = clampedLat * pi / 180.0;
    final x = ((lon + 180.0) / 360.0 * n).floor();
    final y = ((1.0 - log(tan(latRad) + 1.0 / cos(latRad)) / pi) / 2.0 * n).floor();
    return TilePosition(x: x, y: y, zoom: zoom);
  }

  /// Returns the NW corner of a tile as (lat, lon).
  static ({double lat, double lon}) tileToLatLon({
    required int x,
    required int y,
    required int zoom,
  }) {
    final n = pow(2.0, zoom).toDouble();
    final lon = x / n * 360.0 - 180.0;
    final latRad = atan(_sinh(pi * (1.0 - 2.0 * y / n)));
    return (lat: latRad * 180.0 / pi, lon: lon);
  }

  static double _sinh(double x) =&gt; (exp(x) - exp(-x)) / 2.0;
}

/// Bytewise OR of two equal-length bitmaps.
///
/// Returns a NEW [Uint8List] — inputs are not mutated. Monotone by
/// construction: `result[i] &gt;= a[i]` for every byte, because `a | b &gt;= a`.
///
/// Throws [ArgumentError] if lengths differ.
Uint8List mergeBitmap(Uint8List current, Uint8List mask) {
  if (current.length != mask.length) {
    throw ArgumentError.value(
      mask,
      'mask',
      'length ${mask.length} != current length ${current.length}',
    );
  }
  final result = Uint8List(current.length);
  for (var i = 0; i &lt; current.length; i++) {
    result[i] = current[i] | mask[i];
  }
  return result;
}

/// Population count (number of set bits) in [bytes].
///
/// Uses the classic SWAR popcount trick per byte — 4 ops, O(n) in bytes.
int popcount(Uint8List bytes) {
  var count = 0;
  for (final b in bytes) {
    var v = b;
    v = v - ((v &gt;&gt; 1) &amp; 0x55);
    v = (v &amp; 0x33) + ((v &gt;&gt; 2) &amp; 0x33);
    count += (v + (v &gt;&gt; 4)) &amp; 0x0F;
  }
  return count;
}

import 'package:meta/meta.dart';
import '../errors/import_errors.dart';

/// Versioned JSON envelope for import/export (decision D9).
///
/// Shape: `{ schemaVersion: int, type: string, payload: {...} }`. Every
/// shareable document (session, bundle, markers-only, mirk-style) wraps its
/// body in this envelope so [JsonMigrator] can read [schemaVersion] and
/// dispatch migrations before handing [payload] to the entity factory.
@immutable
class Envelope {
  const Envelope({
    required this.schemaVersion,
    required this.type,
    required this.payload,
  });

  /// Schema version as written by the originating instance. Field is on the
  /// envelope (not the payload) so even a future unknown [type] can be read
  /// enough to emit a clear error.
  final int schemaVersion;

  /// Envelope payload discriminator (`'session'`, `'bundle'`, `'markers_only'`,
  /// `'mirk_style'`, ...).
  final String type;

  /// Opaque payload — shape depends on [type] and is validated by the
  /// type-specific factory downstream.
  final Map&lt;String, Object?&gt; payload;

  /// Parses [json] into an [Envelope]. Throws [ImportValidationException] on
  /// missing or malformed required keys.
  factory Envelope.fromJson(Map&lt;String, Object?&gt; json) {
    final rawVersion = json['schemaVersion'];
    if (rawVersion is! int) {
      throw ImportValidationException(
        reason: 'envelope: schemaVersion must be an int, got ${rawVersion?.runtimeType}',
      );
    }
    final rawType = json['type'];
    if (rawType is! String || rawType.isEmpty) {
      throw ImportValidationException(
        reason: 'envelope: type must be a non-empty string',
      );
    }
    final rawPayload = json['payload'];
    if (rawPayload is! Map) {
      throw ImportValidationException(
        reason: 'envelope: payload must be an object, got ${rawPayload?.runtimeType}',
      );
    }
    return Envelope(
      schemaVersion: rawVersion,
      type: rawType,
      payload: Map&lt;String, Object?&gt;.from(rawPayload),
    );
  }

  Map&lt;String, Object?&gt; toJson() =&gt; {
        'schemaVersion': schemaVersion,
        'type': type,
        'payload': payload,
      };

  @override
  bool operator ==(Object other) =&gt;
      other is Envelope &amp;&amp;
      other.schemaVersion == schemaVersion &amp;&amp;
      other.type == type &amp;&amp;
      _mapEquals(other.payload, payload);

  @override
  int get hashCode =&gt; Object.hash(schemaVersion, type, _mapHash(payload));
}

// --- helpers ---
bool _mapEquals(Map&lt;String, Object?&gt; a, Map&lt;String, Object?&gt; b) {
  if (a.length != b.length) return false;
  for (final key in a.keys) {
    if (!b.containsKey(key) || b[key] != a[key]) return false;
  }
  return true;
}

int _mapHash(Map&lt;String, Object?&gt; m) =&gt;
    Object.hashAllUnordered(m.entries.map((e) =&gt; Object.hash(e.key, e.value)));

import '../errors/migration_errors.dart';
import 'json_migration.dart';

/// Chain-of-migrations executor (decision D9).
///
/// Given a list of [JsonMigration] steps (one per version transition), runs
/// them sequentially to migrate a payload from any registered source version
/// to any registered target version.
class JsonMigrator {
  JsonMigrator(List&lt;JsonMigration&gt; migrations)
      : _migrations = List.unmodifiable(migrations);

  final List&lt;JsonMigration&gt; _migrations;

  /// Migrates [payload] from [fromVersion] to [toVersion] applying each
  /// registered step in order.
  ///
  /// Throws [MigrationFailureException] if a step is missing for some
  /// v_n → v_n+1 transition in the range.
  Map&lt;String, Object?&gt; migrate({
    required int fromVersion,
    required int toVersion,
    required Map&lt;String, Object?&gt; payload,
  }) {
    if (toVersion &lt; fromVersion) {
      throw MigrationFailureException(
        reason: 'downgrade not supported (from=$fromVersion, to=$toVersion)',
      );
    }
    var current = payload;
    var v = fromVersion;
    while (v &lt; toVersion) {
      final step = _migrations.where((m) =&gt; m.fromVersion == v).toList();
      if (step.isEmpty) {
        throw MigrationFailureException(
          reason: 'no migrator registered for v$v → v${v + 1}',
        );
      }
      if (step.length &gt; 1) {
        throw MigrationFailureException(
          reason: 'multiple migrators registered for v$v → v${v + 1}',
        );
      }
      current = step.single.apply(current);
      v++;
    }
    return current;
  }
}
