```dart
import 'package:drift/drift.dart';
import 'package:mirkfall/config/constants.dart';

/// Applies the runtime PRAGMAs expected by MirkFall on every DB connection
/// open. Called from [AppDatabase.migration] in the `beforeOpen` callback.
///
/// Pragmas (CONTEXT.md §Stockage DB):
/// - `synchronous = NORMAL` — balances crash safety vs fsync cost for WAL.
/// - `busy_timeout = ${kDbBusyTimeoutMs}` — retry window before `SQLITE_BUSY`.
/// - `foreign_keys = ON` — CRITICAL. Default is OFF, which silently voids
///   the `ON DELETE CASCADE` declarations (RESEARCH pitfall #1).
///
/// NOTE: `journal_mode = WAL` is NOT applied here — it must run on the raw
/// sqlite3 handle BEFORE Drift's first query, via the `setup:` callback of
/// `NativeDatabase.memory`/`createInBackground` (pitfall #2). `beforeOpen`
/// runs after the first connection is already established.
Future&lt;void&gt; applyRuntimePragmas(QueryExecutor executor) async {
  await executor.runCustom('PRAGMA synchronous = NORMAL');
  await executor.runCustom('PRAGMA busy_timeout = $kDbBusyTimeoutMs');
  await executor.runCustom('PRAGMA foreign_keys = ON');
}
```

```dart

import 'dart:convert';
import 'package:drift/drift.dart';
import 'package:mirkfall/domain/mirk/mirk_style_config.dart';
import 'package:mirkfall/domain/sessions/session_status.dart';

class UnixMsToDateTimeConverter extends TypeConverter&lt;DateTime, int&gt; {
  const UnixMsToDateTimeConverter();

  @override
  DateTime fromSql(int fromDb) =&gt;
      DateTime.fromMillisecondsSinceEpoch(fromDb, isUtc: true);

  @override
  int toSql(DateTime value) =&gt; value.toUtc().millisecondsSinceEpoch;
}

class SessionStatusStringConverter extends TypeConverter&lt;SessionStatus, String&gt; {
  const SessionStatusStringConverter();

  @override
  SessionStatus fromSql(String fromDb) =&gt; switch (fromDb) {
        'active' =&gt; SessionStatus.active,
        'stopped' =&gt; SessionStatus.stopped,
        _ =&gt; throw ArgumentError.value(fromDb, 'status', 'unknown SessionStatus'),
      };

  @override
  String toSql(SessionStatus value) =&gt; switch (value) {
        SessionStatus.active =&gt; 'active',
        SessionStatus.stopped =&gt; 'stopped',
      };
}

class MirkStyleConfigJsonConverter extends TypeConverter&lt;MirkStyleConfig, String&gt; {
  const MirkStyleConfigJsonConverter();

  @override
  MirkStyleConfig fromSql(String fromDb) {
    final raw = jsonDecode(fromDb) as Map&lt;String, Object?&gt;;
    return MirkStyleConfig.fromJson(raw);
  }

  @override
  String toSql(MirkStyleConfig value) =&gt; jsonEncode(value.toJson());
}
```

```dart
import 'package:drift/drift.dart';
```

```dart
/// V1 → V2 fictive migration (Phase 03 proof-of-framework).
///
/// Shape change: adds nullable `notes TEXT` column to `t_sessions`. Purpose =
/// exercise the Drift migration tooling (`SchemaVerifier`, `addColumn`) before
/// a real schema change lands. Explicitly symbolic — `notes` has no product
/// feature backing it in V1.0; it exists to prove the pipeline works end-to-end.
///
/// Phase 03 SC#6 + CONTEXT.md §Framework migration.
class V1ToV2Notes {
  V1ToV2Notes._();

  static Future&lt;void&gt; apply(Migrator m, int from, int to) async {
    if (from &lt; 2 &amp;&amp; to &gt;= 2) {
      // `m.addColumn(table, column)` emits `ALTER TABLE t_sessions ADD COLUMN notes TEXT`
      // for SQLite. Drift resolves the column definition from the schema at V2.
      await m.addColumn(m.database.resolvedEngine.attachedDatabase._$AppDatabaseTableTypedTable.sessions, m.database.resolvedEngine.attachedDatabase._$AppDatabaseTableTypedTable.sessions.notes);
    }
  }
}
```

NOTE: The exact Drift API for `m.addColumn` takes a `TableInfo` + `GeneratedColumn` pair. The snippet above is illustrative — the executor must consult the generated `_$AppDatabase` accessors to pass the right references. Simpler form that works across Drift 2.x versions:

```dart
static Future&lt;void&gt; apply(Migrator m, int from, int to, AppDatabase db) async {
  if (from &lt; 2 &amp;&amp; to &gt;= 2) {
    await m.addColumn(db.sessions, db.sessions.notes);
  }
}
```

— but this requires `onUpgrade` in `AppDatabase` to pass `this` through, which adds coupling. Preferred form uses a `customStatement` directly, bypassing Drift's column-reference machinery:

```dart
static Future&lt;void&gt; apply(Migrator m, int from, int to) async {
  if (from &lt; 2 &amp;&amp; to &gt;= 2) {
    await m.database.customStatement(
      'ALTER TABLE t_sessions ADD COLUMN notes TEXT',
    );
  }
}
```

Both approaches are valid. The `customStatement` form is the most portable, survives schema refactors, and is trivially testable via SchemaVerifier. **Recommend the customStatement form.** Document this choice in SUMMARY.md.
