```dart
// lib/infrastructure/stores/drift_session_store.dart
class DriftSessionStore implements SessionStore {
  DriftSessionStore(this._db, this._idGenerator);
  final AppDatabase _db;
  final IdGenerator _idGenerator; // only used for `insert` if caller passes id: null in future — not essential for Phase 03

  @override
  Future&lt;List&lt;Session&gt;&gt; listAll() =&gt; _db.select(_db.sessions).get().then((rows) =&gt; rows.map(_hydrate).toList());

  @override
  Future&lt;Session?&gt; findById(SessionId id) async {
    final row = await (_db.select(_db.sessions)..where((t) =&gt; t.id.equals(id.value))).getSingleOrNull();
    return row == null ? null : _hydrate(row);
  }

  @override
  Future&lt;Session&gt; requireById(SessionId id) async {
    final session = await findById(id);
    if (session == null) throw SessionNotFoundException(id: id);
    return session;
  }

  @override
  Future&lt;Session?&gt; findActive() async {
    final row = await (_db.select(_db.sessions)..where((t) =&gt; t.status.equals('active'))).getSingleOrNull();
    return row == null ? null : _hydrate(row);
  }

  @override
  Future&lt;void&gt; insert(Session session) async =&gt; _db.into(_db.sessions).insert(_toCompanion(session));

  @override
  Future&lt;void&gt; update(Session session) async {
    await (_db.update(_db.sessions)..where((t) =&gt; t.id.equals(session.id.value)))
        .write(_toCompanion(session));
  }

  @override
  Future&lt;void&gt; delete(SessionId id) async {
    await (_db.delete(_db.sessions)..where((t) =&gt; t.id.equals(id.value))).go();
  }

  @override
  Future&lt;void&gt; activate(SessionId id) async {
    try {
      await (_db.update(_db.sessions)..where((t) =&gt; t.id.equals(id.value)))
          .write(const SessionsCompanion(status: Value('active')));
    } on SqliteException catch (e) {
      if (e.extendedResultCode == 2067) {
        throw ConcurrentActivationException(attemptedId: id);
      }
      rethrow;
    }
  }

  @override
  Future&lt;void&gt; deactivate(SessionId id) async {
    await (_db.update(_db.sessions)..where((t) =&gt; t.id.equals(id.value)))
        .write(const SessionsCompanion(status: Value('stopped')));
  }

  // -- hydration --
  Session _hydrate(SessionData row) =&gt; Session(
    id: SessionId(row.id),
    displayName: row.displayName,
    status: SessionStatusStringConverter().fromSql(row.status),
    startedAtUtc: row.startedAtUtc,
    startedAtOffsetMinutes: row.startedAtOffsetMinutes,
    stoppedAtUtc: row.stoppedAtUtc,
    stoppedAtOffsetMinutes: row.stoppedAtOffsetMinutes,
    notes: row.notes,
  );

  SessionsCompanion _toCompanion(Session s) =&gt; SessionsCompanion.insert(
    id: s.id.value,
    displayName: s.displayName,
    status: SessionStatusStringConverter().toSql(s.status),
    startedAtUtc: s.startedAtUtc,
    startedAtOffsetMinutes: s.startedAtOffsetMinutes,
    stoppedAtUtc: Value(s.stoppedAtUtc),
    stoppedAtOffsetMinutes: Value(s.stoppedAtOffsetMinutes),
    notes: Value(s.notes),
  );
}
```

```dart
class DriftRevealedTileStore implements RevealedTileStore {
  DriftRevealedTileStore(this._db, this._idGenerator);
  final AppDatabase _db;
  final IdGenerator _idGenerator;

  @override
  Future&lt;List&lt;RevealedTile&gt;&gt; listBySession(SessionId sessionId) async {
    final rows = await (_db.select(_db.revealedTiles)
          ..where((t) =&gt; t.sessionId.equals(sessionId.value)))
        .get();
    return rows.map(_hydrate).toList();
  }

  @override
  Future&lt;RevealedTile?&gt; findByParent({
    required SessionId sessionId, required int parentX, required int parentY,
  }) async {
    final row = await (_db.select(_db.revealedTiles)
          ..where((t) =&gt;
              t.sessionId.equals(sessionId.value) &amp;
              t.parentX.equals(parentX) &amp;
              t.parentY.equals(parentY)))
        .getSingleOrNull();
    return row == null ? null : _hydrate(row);
  }

  @override
  Future&lt;void&gt; mergeMask({
    required SessionId sessionId,
    required int parentX, required int parentY,
    required Uint8List mask,
  }) async {
    await _db.transaction(() async {
      final existing = await (_db.select(_db.revealedTiles)
        ..where((t) =&gt;
            t.sessionId.equals(sessionId.value) &amp;
            t.parentX.equals(parentX) &amp;
            t.parentY.equals(parentY))).getSingleOrNull();
      final now = DateTime.now().toUtc();
      if (existing == null) {
        await _db.into(_db.revealedTiles).insert(RevealedTilesCompanion.insert(
          id: _idGenerator.newId('rvt_'),
          sessionId: sessionId.value,
          parentX: parentX,
          parentY: parentY,
          bitmap: mask,
          setBitCount: Value(popcount(mask)),
          updatedAtUtc: now,
        ));
      } else {
        final merged = mergeBitmap(existing.bitmap, mask);
        await (_db.update(_db.revealedTiles)..where((t) =&gt; t.id.equals(existing.id)))
          .write(RevealedTilesCompanion(
            bitmap: Value(merged),
            setBitCount: Value(popcount(merged)),
            updatedAtUtc: Value(now),
          ));
      }
    });
  }

  RevealedTile _hydrate(RevealedTileData row) =&gt; RevealedTile(
    id: RevealedTileId(row.id),
    sessionId: SessionId(row.sessionId),
    parentX: row.parentX,
    parentY: row.parentY,
    parentZoom: row.parentZoom,
    bitmap: row.bitmap,
    setBitCount: row.setBitCount,
    updatedAtUtc: row.updatedAtUtc,
  );
}
```

```dart
      throwsA(isA&lt;ConcurrentActivationException&gt;()

    expect(active?.id, id1);
```

```dart

// See LICENSE file for details
import 'dart:typed_data';
import 'package:drift/drift.dart';
import 'package:drift/native.dart';
import 'package:mirkfall/config/constants.dart';
import 'package:mirkfall/domain/ids/session_id.dart';

import 'package:test/test.dart';

void main() {
  late AppDatabase db;
  late DriftRevealedTileStore store;
  final sessionId = SessionId('sess_01HRTESTIDEMPOTENCEAAAAAAAA');

  setUp(() async {
    db = AppDatabase(DatabaseConnection(
      NativeDatabase.memory(setup: (r) =&gt; r.execute('PRAGMA journal_mode = WAL')),
      closeStreamsSynchronously: true,
    ));
    store = DriftRevealedTileStore(db, SeededIdGenerator(seed: 42));
    // seed session
    await db.customStatement('''
      INSERT INTO t_sessions (id, display_name, status, started_at_utc, started_at_offset_minutes)
      VALUES ('${sessionId.value}', 'T', 'stopped', 1000, 120)
    ''');
  });
  tearDown(() async =&gt; db.close());

  Uint8List _mask(List&lt;int&gt; setBytes) {
    final m = Uint8List(kRevealedTileBitmapBytes);
    for (final i in setBytes) m[i] = 0xFF;
    return m;
  }

  test('applying same mask twice → idempotent (byte-equal bitmap, same setBitCount)', () async {
    final mask = _mask([0, 1, 2]);
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: mask);
    final first = await store.findByParent(sessionId: sessionId, parentX: 1, parentY: 1);
    expect(first, isNotNull);
    final bitmapFirst = first!.bitmap;
    final countFirst = first.setBitCount;

    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: mask);
    final second = await store.findByParent(sessionId: sessionId, parentX: 1, parentY: 1);
    expect(second!.bitmap, bitmapFirst);
    expect(second.setBitCount, countFirst);
  });

  test('additive mask B on top of A → final bitmap == A | B', () async {
    final a = _mask([0, 1, 2]);       // bits in bytes 0..2
    final b = _mask([3, 4, 5]);       // bits in bytes 3..5
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: a);
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: b);
    final row = await store.findByParent(sessionId: sessionId, parentX: 1, parentY: 1);
    expect(row, isNotNull);
    for (var i = 0; i &lt; kRevealedTileBitmapBytes; i++) {
      expect(row!.bitmap[i], a[i] | b[i], reason: 'byte $i');
    }
    expect(row!.setBitCount, 48); // 6 bytes × 8 bits
  });

  test('monotone: mask of all-zeros preserves existing bits', () async {
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1,
        mask: _mask([0, 1, 2]));
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1,
        mask: Uint8List(kRevealedTileBitmapBytes));
    final row = await store.findByParent(sessionId: sessionId, parentX: 1, parentY: 1);
    expect(row!.setBitCount, 24); // 3 bytes × 8 bits
  });

  test('mergeMask rejects wrong-size mask', () async {
    expect(
      () =&gt; store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: Uint8List(100)),
      throwsA(isA&lt;ArgumentError&gt;()),
    );
  });

  test('first mergeMask creates a row; subsequent maintain a single row per (sess,x,y)', () async {
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: _mask([0]));
    await store.mergeMask(sessionId: sessionId, parentX: 1, parentY: 1, mask: _mask([1]));
    final rows = await store.listBySession(sessionId);
    expect(rows.where((r) =&gt; r.parentX == 1 &amp;&amp; r.parentY == 1), hasLength(1));
  });
}
```

```dart
  final supportDir = await getApplicationSupportDirectory();
  final dbFilename = p.join(supportDir.path, kDbFilename);
```
