Exit 0 if clean, exit 1 on violation (with file:line context), exit 2 on misconfiguration (e.g., lib/domain/ missing). Matches Phase 01 exit-code contract (0/1/2).

Implementation sketch (~60 lines):

```dart
// Copyright (c) 2026 THONGVAN Alexis
// Licensed under the Good Old Software License v1.0
// See LICENSE file for details

import 'dart:io';
import 'package:path/path.dart' as p;

/// Forbidden-import scanner for `lib/domain/`.
///
/// Domain purity invariant (Phase 03 SC#4): zero imports of `package:flutter`
/// or `package:drift` under `lib/domain/`. Application + infrastructure layers
/// may freely use them; domain must stay pure Dart so `dart test` runs
/// without a Flutter toolchain.
///
/// Exit codes: 0 = clean, 1 = violations found, 2 = misconfiguration.
Future<void> main(List<String> args) async {
  final domainDir = Directory(p.join(Directory.current.path, 'lib', 'domain'));
  if (!domainDir.existsSync()) {
    stderr.writeln('error: lib/domain/ not found at ${domainDir.path}');
    exit(2);
  }

  final violations = <String>[];
  final forbiddenPattern = RegExp(
    r'''^\s*import\s+['"]package:(flutter|drift)(?:/|['"])''',
    multiLine: false,
  );

  await for (final entity in domainDir.list(recursive: true, followLinks: false)) {
    if (entity is! File) continue;
    if (!entity.path.endsWith('.dart')) continue;
    if (entity.path.endsWith('.g.dart') || entity.path.endsWith('.freezed.dart')) {
      continue; // generated files — not in source control of domain purity
    }
    final lines = await entity.readAsLines();
    for (var i = 0; i < lines.length; i++) {
      if (forbiddenPattern.hasMatch(lines[i])) {
        violations.add('${p.relative(entity.path)}:${i + 1}: ${lines[i].trim()}');
      }
    }
  }

  if (violations.isEmpty) {
    stdout.writeln('domain purity OK: ${_countDartFiles(domainDir)} files scanned, zero forbidden imports');
    exit(0);
  }

  stderr.writeln('DOMAIN PURITY VIOLATION: ${violations.length} forbidden import(s):');
  for (final v in violations) {
    stderr.writeln('  $v');
  }
  stderr.writeln('Rule: lib/domain/ must not import package:flutter/* or package:drift/*.');
  stderr.writeln('Move the offending import to lib/application/ or lib/infrastructure/.');
  exit(1);
}

int _countDartFiles(Directory dir) {
  var count = 0;
  for (final e in dir.listSync(recursive: true, followLinks: false)) {
    if (e is File &amp;&amp;
        e.path.endsWith('.dart') &amp;&amp;
        !e.path.endsWith('.g.dart') &amp;&amp;
        !e.path.endsWith('.freezed.dart')) {
      count++;
    }
  }
  return count;
}
```
