```dart
// Copyright (c) 2026 THONGVAN Alexis

import 'dart:io';

import 'package:path/path.dart' as p;

/// Copies the MirkFall SQLite DB file on demand, keeping a rolling window of
/// the N most recent backups.
///
/// Phase 03 use: pre-migration safety net (CONTEXT.md §Timing backups).
/// Phase 15 use: "Backup DB now" button in the debug menu (optional).
class DbBackupService {
  DbBackupService({
    required this.dbFilename,
    required this.backupsDir,
    required this.maxBackups,
    DateTime Function()? clock,
  }) : _clock = clock ?? DateTime.now;

  /// Absolute path to the live DB file.
  final String dbFilename;

  /// Directory where backups are stored.
  final Directory backupsDir;

  /// Rolling retention limit.
  final int maxBackups;

  final DateTime Function() _clock;

  /// Copies [dbFilename] to a new file in [backupsDir], naming it
  /// `mirkfall.db.backup-v{from}-to-v{to}-{iso-utc}`. Then rotates.
  Future&lt;File&gt; takeBackup({required int fromVersion, required int toVersion}) async {
    if (!await backupsDir.exists()) {
      await backupsDir.create(recursive: true);
    }
    // ISO 8601 UTC, colons replaced with hyphens for cross-platform filename safety.
    final timestamp = _clock().toUtc().toIso8601String().replaceAll(':', '-');
    final backupBasename = 'mirkfall.db.backup-v$fromVersion-to-v$toVersion-$timestamp';
    final backupFilename = p.join(backupsDir.path, backupBasename);

    final source = File(dbFilename);
    final target = await source.copy(backupFilename);
    await rotate();
    return target;
  }

  /// Deletes the oldest files in [backupsDir] so at most [maxBackups] remain.
  Future&lt;void&gt; rotate() async {
    if (!await backupsDir.exists()) return;
    final entries = await backupsDir
        .list(followLinks: false)
        .where((e) =&gt; e is File)
        .cast&lt;File&gt;()
        .toList();
    if (entries.length &lt;= maxBackups) return;
    entries.sort((a, b) =&gt; b.statSync().modified.compareTo(a.statSync().modified));
    for (final file in entries.skip(maxBackups)) {
      await file.delete();
    }
  }
}
```

```dart
// Copyright (c) 2026 THONGVAN Alexis

import 'package:drift/drift.dart';

import '../../domain/errors/migration_errors.dart';

/// Captures per-table row counts and hard-fails on any post-migration loss.
///
/// RESEARCH pitfall #7: `SchemaVerifier.migrateAndValidate` verifies schema
/// shape only. Data survival must be checked independently — which is what
/// this class does.
class SchemaSanityChecker {
  const SchemaSanityChecker(this._executor);

  final QueryExecutor _executor;

  static const List&lt;String&gt; _tables = [
    't_sessions',
    't_markers',
    't_revealed_tiles',
    't_marker_categories',
    't_mirk_styles',
    't_photos',
  ];

  /// Returns row-count per MirkFall table.
  Future&lt;Map&lt;String, int&gt;&gt; captureRowCounts() async {
    final result = &lt;String, int&gt;{};
    for (final table in _tables) {
      final row = await _executor.runSelect('SELECT COUNT(*) AS c FROM $table', const []);
      result[table] = (row.first['c'] as int);
    }
    return result;
  }

  /// Throws [MigrationFailureException] if any table's row count decreased.
  ///
  /// Growth is silently accepted (an `onUpgrade` step legitimately adding
  /// seed rows would increase counts — e.g., seeding a default category).
  void assertNoLoss(Map&lt;String, int&gt; before, Map&lt;String, int&gt; after) {
    for (final entry in before.entries) {
      final afterCount = after[entry.key] ?? 0;
      if (afterCount &lt; entry.value) {
        throw MigrationFailureException(
          reason: 'row count decreased on ${entry.key}: '
              '${entry.value} → $afterCount (migration likely dropped data)',
        );
      }
    }
  }
}
```

```dart
import 'dart:io';

import 'package:mirkfall/infrastructure/db/backup.dart';
import 'package:path/path.dart' as p;
import 'package:test/test.dart';

void main() {
  late Directory scratchDir;
  late File fakeDb;
  late Directory backupsDir;

  setUp(() {
    scratchDir = Directory.systemTemp.createTempSync('mirkfall_backup_test_');
    fakeDb = File(p.join(scratchDir.path, 'mirkfall.db'));
    fakeDb.writeAsBytesSync(List.generate(128, (i) =&gt; i &amp; 0xFF));
    backupsDir = Directory(p.join(scratchDir.path, 'db_backups'));
  });

  tearDown(() {
    if (scratchDir.existsSync()) {
      scratchDir.deleteSync(recursive: true);
    }
  });

  test('takeBackup creates a file with correctly-formatted name', () async {
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir,
      maxBackups: 3,
      clock: () =&gt; DateTime.utc(2026, 4, 18, 12, 30, 45, 123),
    );

    final backup = await svc.takeBackup(fromVersion: 1, toVersion: 2);

    expect(backup.existsSync(), isTrue);
    expect(
      p.basename(backup.path),
      equals('mirkfall.db.backup-v1-to-v2-2026-04-18T12-30-45.123Z'),
    );
    // Byte-equal content
    expect(backup.readAsBytesSync(), fakeDb.readAsBytesSync());
  });

  test('rotate keeps the 3 newest when 4 exist', () async {
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir,
      maxBackups: 3,
    );
    backupsDir.createSync(recursive: true);

    final now = DateTime.now();
    for (var i = 0; i &lt; 4; i++) {
      final f = File(p.join(backupsDir.path, 'mirkfall.db.backup-v1-to-v2-fake-$i'));
      f.writeAsBytesSync([i]);
      // Stagger modified times
      f.setLastModifiedSync(now.subtract(Duration(hours: 4 - i)));
    }

    await svc.rotate();

    final remaining = backupsDir.listSync().whereType&lt;File&gt;().toList();
    expect(remaining.length, 3);
    // The one with the oldest mtime ("fake-0") should have been deleted.
    final names = remaining.map((f) =&gt; p.basename(f.path)).toSet();
    expect(names.contains('mirkfall.db.backup-v1-to-v2-fake-0'), isFalse);
    expect(names.contains('mirkfall.db.backup-v1-to-v2-fake-1'), isTrue);
    expect(names.contains('mirkfall.db.backup-v1-to-v2-fake-2'), isTrue);
    expect(names.contains('mirkfall.db.backup-v1-to-v2-fake-3'), isTrue);
  });

  test('rotate is a no-op when fewer backups than maxBackups', () async {
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir,
      maxBackups: 3,
    );
    backupsDir.createSync(recursive: true);
    File(p.join(backupsDir.path, 'only.one')).writeAsBytesSync([1]);

    await svc.rotate();

    expect(backupsDir.listSync().length, 1);
  });

  test('rotate handles missing dir silently', () async {
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir, // does not exist
      maxBackups: 3,
    );
    // Should not throw.
    await svc.rotate();
  });

  test('takeBackup creates backupsDir if absent', () async {
    expect(backupsDir.existsSync(), isFalse);
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir,
      maxBackups: 3,
      clock: () =&gt; DateTime.utc(2026, 4, 18),
    );
    await svc.takeBackup(fromVersion: 1, toVersion: 2);
    expect(backupsDir.existsSync(), isTrue);
  });

  test('consecutive takeBackup calls + rotation keeps at most maxBackups', () async {
    final svc = DbBackupService(
      dbFilename: fakeDb.path,
      backupsDir: backupsDir,
      maxBackups: 3,
    );
    // 5 backups, spaced 1 ms apart (real clock OK — we don't check names here)
    for (var i = 0; i &lt; 5; i++) {
      await svc.takeBackup(fromVersion: 1, toVersion: 2);
      await Future&lt;void&gt;.delayed(const Duration(milliseconds: 5));
    }
    final remaining = backupsDir.listSync().whereType&lt;File&gt;().toList();
    expect(remaining.length, 3);
  });
}
```

```dart
import 'package:drift/drift.dart';
import 'package:drift/native.dart';
import 'package:mirkfall/domain/errors/migration_errors.dart';
import 'package:mirkfall/infrastructure/db/app_database.dart';
import 'package:mirkfall/infrastructure/db/schema_sanity.dart';
import 'package:test/test.dart';

void main() {
  late AppDatabase db;

  setUp(() {
    db = AppDatabase(
      DatabaseConnection(
        NativeDatabase.memory(setup: (r) =&gt; r.execute('PRAGMA journal_mode = WAL')),
        closeStreamsSynchronously: true,
      ),
    );
  });
  tearDown(() async =&gt; db.close());

  test('captureRowCounts returns all 6 tables', () async {
    await db.customStatement('SELECT 1'); // ensure DB opened (pragmas applied)
    final checker = SchemaSanityChecker(db.executor);
    final counts = await checker.captureRowCounts();
    expect(counts.keys, containsAll([
      't_sessions', 't_markers', 't_revealed_tiles',
      't_marker_categories', 't_mirk_styles', 't_photos',
    ]));
    // Fresh DB — all zero.
    for (final entry in counts.entries) {
      expect(entry.value, 0, reason: '${entry.key} should be empty on fresh DB');
    }
  });

  test('captureRowCounts reflects INSERTs', () async {
    await db.customStatement('''
      INSERT INTO t_marker_categories (id, display_name, icon_name, created_at_utc, created_at_offset_minutes)
      VALUES ('cat_default', 'Default', 'pin', 1000, 120)
    ''');
    await db.customStatement('''
      INSERT INTO t_sessions (id, display_name, status, started_at_utc, started_at_offset_minutes)
      VALUES ('sess_S1', 'S1', 'stopped', 1000, 120)
    ''');
    final checker = SchemaSanityChecker(db.executor);
    final counts = await checker.captureRowCounts();
    expect(counts['t_marker_categories'], 1);
    expect(counts['t_sessions'], 1);
    expect(counts['t_markers'], 0);
  });

  test('assertNoLoss: identical counts → silent', () {
    final checker = SchemaSanityChecker(db.executor);
    final before = {'t_sessions': 10, 't_markers': 50};
    final after  = {'t_sessions': 10, 't_markers': 50};
    checker.assertNoLoss(before, after); // must not throw
  });

  test('assertNoLoss: growth → silent', () {
    final checker = SchemaSanityChecker(db.executor);
    final before = {'t_sessions': 10, 't_marker_categories': 2};
    final after  = {'t_sessions': 10, 't_marker_categories': 3}; // seeded +1 by onUpgrade
    checker.assertNoLoss(before, after);
  });

  test('assertNoLoss: loss → MigrationFailureException mentions table', () {
    final checker = SchemaSanityChecker(db.executor);
    final before = {'t_sessions': 10};
    final after  = {'t_sessions': 9};
    expect(
      () =&gt; checker.assertNoLoss(before, after),
      throwsA(isA&lt;MigrationFailureException&gt;().having(
          (e) =&gt; e.reason, 'reason', contains('t_sessions'))),
    );
  });

  test('assertNoLoss: missing table key → treated as 0 → throws if before was non-zero', () {
    final checker = SchemaSanityChecker(db.executor);
    final before = {'t_markers': 50};
    final after  = &lt;String, int&gt;{}; // no entry
    expect(
      () =&gt; checker.assertNoLoss(before, after),
      throwsA(isA&lt;MigrationFailureException&gt;()),
    );
  });
}
```

```dart
import 'package:mirkfall/infrastructure/db/app_database.dart';
import 'package:mirkfall/infrastructure/db/schema_sanity.dart';
import 'package:path/path.dart' as p;
import 'package:test/test.dart';

import '../../generated_migrations/schema.dart';

void main() {
  late SchemaVerifier verifier;
  setUpAll(() =&gt; verifier = SchemaVerifier(GeneratedHelper()));

  group('V1→V2 migration (notes column)', () {
    test('single session row preserved, notes defaults to NULL, column is writeable', () async {
      // ── Setup: open at V1 schema + insert a session row in V1 shape.
      final schema = await verifier.schemaAt(1);

      final rawInsert = DatabaseConnection(schema.newConnection(), closeStreamsSynchronously: true);
      await rawInsert.runCustom(
        "INSERT INTO t_sessions (id, display_name, status, started_at_utc, started_at_offset_minutes) "
        "VALUES ('sess_01HRMIGRATIONTESTAAAAAAAAA', 'Paris trip', 'stopped', 1712000000000, 120)",
      );
      await rawInsert.close();

      // ── Run migration V1→V2 via prod AppDatabase wired to the same schema file.
      final prodDb = AppDatabase(schema.newConnection());
      await verifier.migrateAndValidate(prodDb, 2);

      // ── Assert: row survived + notes is NULL + notes is writeable.
      final rows = await prodDb.customSelect(
        "SELECT id, display_name, status, started_at_utc, started_at_offset_minutes, notes "
        "FROM t_sessions WHERE id = 'sess_01HRMIGRATIONTESTAAAAAAAAA'",
      ).get();
      expect(rows, hasLength(1));
      final row = rows.single;
      expect(row.read&lt;String&gt;('id'), 'sess_01HRMIGRATIONTESTAAAAAAAAA');
      expect(row.read&lt;String&gt;('display_name'), 'Paris trip');
      expect(row.read&lt;String&gt;('status'), 'stopped');
      expect(row.read&lt;int&gt;('started_at_utc'), 1712000000000);
      expect(row.read&lt;int&gt;('started_at_offset_minutes'), 120);
      expect(row.readNullable&lt;String&gt;('notes'), isNull);

      // Writeable check
      await prodDb.customStatement(
        "UPDATE t_sessions SET notes = 'Added after migration' "
        "WHERE id = 'sess_01HRMIGRATIONTESTAAAAAAAAA'",
      );
      final reread = await prodDb.customSelect(
        "SELECT notes FROM t_sessions WHERE id = 'sess_01HRMIGRATIONTESTAAAAAAAAA'",
      ).getSingle();
      expect(reread.read&lt;String&gt;('notes'), 'Added after migration');

      await prodDb.close();
    });

    test('full v1_baseline.sql fixture (68 rows) survives V1→V2 migration', () async {
      // Load the 03-01 fixture into a V1 schema, migrate to V2, check every
      // row is still present.
      final schema = await verifier.schemaAt(1);
      final seedExecutor = DatabaseConnection(schema.newConnection(), closeStreamsSynchronously: true);

      final sqlFilename = p.join(
        Directory.current.path,
        'test', 'fixtures', 'db_seed', 'v1_baseline.sql',
      );
      final sql = File(sqlFilename).readAsStringSync();

      // Naive split — fixture is hand-controlled, no embedded semicolons in values.
      for (final stmt in sql.split(';').map((s) =&gt; s.trim()).where((s) =&gt; s.isNotEmpty &amp;&amp; !s.startsWith('--'))) {
        await seedExecutor.runCustom(stmt);
      }

      final sanityBefore = SchemaSanityChecker(seedExecutor);
      final before = await sanityBefore.captureRowCounts();
      expect(before['t_sessions'], 10);
      expect(before['t_markers'], 50);
      expect(before['t_revealed_tiles'], 5);
      expect(before['t_marker_categories'], 3);
      expect(before['t_mirk_styles'], 2);
      await seedExecutor.close();

      // Migrate
      final prodDb = AppDatabase(schema.newConnection());
      await verifier.migrateAndValidate(prodDb, 2);

      final sanityAfter = SchemaSanityChecker(prodDb.executor);
      final after = await sanityAfter.captureRowCounts();
      sanityBefore.assertNoLoss(before, after); // must not throw

      // Spot-check: count is identical (we don't expect any onUpgrade to add rows).
      expect(after['t_sessions'], 10);
      expect(after['t_markers'], 50);
      expect(after['t_revealed_tiles'], 5);
      expect(after['t_marker_categories'], 3);
      expect(after['t_mirk_styles'], 2);

      await prodDb.close();
    });
  });
}
```

The exact method names can differ between Drift versions — verify against the actual `drift_dev` 2.32.x API at implementation time.
