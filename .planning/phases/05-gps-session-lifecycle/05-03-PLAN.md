```dart
Future<LocationPermissionOutcome> requestLocationAlways() async {
  final whenInUse = await Permission.locationWhenInUse.request();
  if (whenInUse.isPermanentlyDenied) return LocationPermissionOutcome.permanentlyDenied;
  if (!whenInUse.isGranted) return LocationPermissionOutcome.denied;
  final always = await Permission.locationAlways.request();
  if (always.isGranted) return LocationPermissionOutcome.granted;
  if (always.isPermanentlyDenied) return LocationPermissionOutcome.permanentlyDenied;
  return LocationPermissionOutcome.whileInUseOnly;
}
```

```dart
@Riverpod(keepAlive: true)
class ActiveSessionController extends _$ActiveSessionController {
  StreamSubscription<Fix>? _sub;
  LocationStream? _stream;
  SessionId? _currentSessionId;

  @override
  FutureOr<ActiveSessionState> build() {
    ref.onDispose(() async {
      await _sub?.cancel();
      _sub = null;
      await _stream?.dispose();
      _stream = null;
    });
    return const Idle();
  }

  Future<void> start(SessionId id) async {
    state = const AsyncData(Idle());   // clear prior state
    state = AsyncData(Starting(id));

    try {
      final settings = await ref.read(sessionSettingsProvider.future);
      final sessionStore = await ref.read(sessionStoreProvider.future);
      final fixStore = await ref.read(fixStoreProvider.future);
      final locationStream = await ref.read(locationStreamProvider.future);
      final notificationService =
          await ref.read(sessionNotificationServiceProvider.future);

      final activated = await sessionStore.activate(id);
      await notificationService.initialize();

      _stream = locationStream;
      _currentSessionId = id;

      _sub = locationStream.positions(
        sessionId: id,
        sessionDisplayName: activated.displayName,
        distanceFilterMeters: settings.distanceFilterMeters,
      ).listen(
        (fix) => _onFix(fix, fixStore),
        onError: _onStreamError,
        cancelOnError: false,
      );

      state = AsyncData(Tracking(
        sessionId: id,
        startedAtUtc: activated.startedAtUtc,
        fixCount: 0,
        lastFix: null,
        distanceFilterMeters: settings.distanceFilterMeters,
      ));
    } on GpsError catch (e) {
      state = AsyncData(ErrorState(e));
      rethrow;
    }
    // ConcurrentActivationException is not a GpsError — let it propagate
    // untyped; UI layer catches it and presents the "stop current first" UX.
  }

  Future<void> stop() async {
    await _sub?.cancel();
    _sub = null;
    await _stream?.dispose();
    _stream = null;

    try {
      final notificationService =
          await ref.read(sessionNotificationServiceProvider.future);
      await notificationService.dismiss();
    } catch (_) {
      // Dismiss is best-effort; do not block stop() on it.
    }

    final current = _currentSessionId;
    if (current != null) {
      try {
        final sessionStore = await ref.read(sessionStoreProvider.future);
        await sessionStore.deactivate(current);
      } catch (_) {
        // Session already deactivated (race with another caller); log + swallow.
      }
      _currentSessionId = null;
    }

    state = const AsyncData(Idle());
  }

  Future<void> _onFix(Fix fix, FixStore fixStore) async {
    await fixStore.insert(fix);
    final current = state.valueOrNull;
    if (current is Tracking) {
      state = AsyncData(current.copyWith(
        fixCount: current.fixCount + 1,
        lastFix: fix,
      ));
    }
  }

  void _onStreamError(Object error, StackTrace st) {
    if (error is GpsError) {
      state = AsyncData(ErrorState(error));
    } else {
      // Unexpected — surface as TrackingBackgroundKilled (best-effort typing).
      state = AsyncData(const ErrorState(TrackingBackgroundKilledException()));
    }
  }
}
```
