```python
#!/usr/bin/env python3
# Copyright (c) 2026 THONGVAN Alexis
# Licensed under the Good Old Software License v1.0

"""Plot a MirkFall session's t_fixes on an OSM static map (Phase 05 POC tool).

Standalone Python 3 CLI — does NOT ship in the app binary.
Python deps pinned in tool/requirements.txt; licences in tool/README.md.
"""

import argparse
import os
import sqlite3
import statistics
import sys
from datetime import datetime, timezone
from pathlib import Path

from staticmap import CircleMarker, Line, StaticMap

OSM_TILE_TEMPLATE = "https://tile.openstreetmap.org/{z}/{x}/{y}.png"
USER_AGENT = (
    "MirkFall-POC-Plotter/1.0 "
    "(+https://github.com/saibashirudo/GOSL-MirkFall)"
)


def parse_args() -> argparse.Namespace:
    p = argparse.ArgumentParser(description=__doc__)
    p.add_argument("--db", required=True, help="Path to mirkfall.db")
    p.add_argument("--session-id", required=True, help="Session ID (sess_...)")
    p.add_argument("--out", default=None, help="Output PNG path")
    p.add_argument("--zoom", type=int, default=None, help="OSM zoom level")
    p.add_argument("--width", type=int, default=1600)
    p.add_argument("--height", type=int, default=1200)
    return p.parse_args()


def fetch_fixes(db_path: str, session_id: str) -> list[tuple[float, float, int]]:
    con = sqlite3.connect(db_path)
    try:
        cur = con.execute(
            "SELECT latitude, longitude, recorded_at_utc FROM t_fixes "
            "WHERE session_id = ? ORDER BY recorded_at_utc ASC",
            (session_id,),
        )
        return [(row[0], row[1], row[2]) for row in cur.fetchall()]
    finally:
        con.close()


def print_stats(fixes: list[tuple[float, float, int]]) -> None:
    if len(fixes) < 2:
        print(f"Only {len(fixes)} fix(es) — not enough for stats.")
        return
    timestamps = [f[2] for f in fixes]
    intervals_s = [
        (timestamps[i] - timestamps[i - 1]) / 1000.0
        for i in range(1, len(timestamps))
    ]
    start = datetime.fromtimestamp(timestamps[0] / 1000, tz=timezone.utc)
    end = datetime.fromtimestamp(timestamps[-1] / 1000, tz=timezone.utc)
    duration_min = (timestamps[-1] - timestamps[0]) / 1000.0 / 60.0
    lats = [f[0] for f in fixes]
    lons = [f[1] for f in fixes]

    print(f"Fixes:        {len(fixes)}")
    print(f"Start (UTC):  {start.isoformat()}")
    print(f"End   (UTC):  {end.isoformat()}")
    print(f"Duration:     {duration_min:.1f} min")
    print(f"Interval min: {min(intervals_s):.1f} s")
    print(f"Interval med: {statistics.median(intervals_s):.1f} s")
    print(f"Interval max: {max(intervals_s):.1f} s")
    print(f"Bbox lat:     [{min(lats):.5f}, {max(lats):.5f}]")
    print(f"Bbox lon:     [{min(lons):.5f}, {max(lons):.5f}]")


def render(
    fixes: list[tuple[float, float, int]],
    out_path: str,
    width: int,
    height: int,
    zoom: int | None,
) -> None:
    m = StaticMap(
        width=width,
        height=height,
        url_template=OSM_TILE_TEMPLATE,
        headers={"User-Agent": USER_AGENT},
    )
    coords = [(lon, lat) for (lat, lon, _) in fixes]
    m.add_line(Line(coords, "red", 3))
    if fixes:
        lat0, lon0, _ = fixes[0]
        latn, lonn, _ = fixes[-1]
        m.add_marker(CircleMarker((lon0, lat0), "green", 10))
        m.add_marker(CircleMarker((lonn, latn), "blue", 10))
    image = m.render(zoom=zoom) if zoom is not None else m.render()
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    image.save(out_path)
    print(f"Saved: {out_path}")


def main() -> int:
    args = parse_args()
    if not Path(args.db).is_file():
        print(f"DB not found: {args.db}", file=sys.stderr)
        return 1
    fixes = fetch_fixes(args.db, args.session_id)
    if not fixes:
        print(f"No fixes for session {args.session_id}", file=sys.stderr)
        return 1
    print_stats(fixes)
    out = args.out or (
        f"docs/poc-artifacts/{args.session_id}-"
        f"{datetime.now().strftime('%Y%m%d-%H%M%S')}.png"
    )
    render(fixes, out, args.width, args.height, args.zoom)
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

**## Project description**:
MirkFall is a personal, offline-first travel journal: as you walk or explore, a "fog of war" over a map of the world gradually reveals itself, showing where you have actually been. It is distributed as free open-source software under the Good Old Software License v1.0, available at https://github.com/saibashirudo/GOSL-MirkFall. It has no servers, no advertising, no analytics, no crash reporting, and no in-app purchases. It is a single-developer project intended for enthusiasts who want to own their exploration data on their device.

**## Why Always location is required**:
An exploration session can last many hours. The user typically starts a session, locks the phone, puts it in a pocket, and walks — "continuous background location while screen is off" is the core feature. Without `Always` authorization, iOS / Android suspend location callbacks when the phone is locked, leaving gaps of tens of minutes in the trail and making the session journal unreliable. MirkFall requests `While Using` first and then `Always`, and explains the reason in-app before prompting (pre-prompt rationale screen).

**## Data handling**:
MirkFall is local-only by design. All GPS fixes, session metadata, and optional photos remain in an on-device SQLite database and filesystem. **No data is ever transmitted to any server, no analytics SDK is embedded, no crash reporting is automatically forwarded.** The licence (GOSL v1.0) explicitly prohibits adding telemetry to redistributed or derived builds. Users can export their session data as a JSON+ZIP archive via the OS share sheet — the transport is user-initiated, not automatic.

**## Source code accessibility**:
The full source code is public at https://github.com/saibashirudo/GOSL-MirkFall under GOSL v1.0. Reviewers are welcome to audit the codebase for the claims in this document: in particular, `DEPENDENCIES.md` at the repo root lists every third-party library with its licence and a telemetry audit entry (confirmed zero analytics SDKs).

Developer email: saibashirudo@protonmail.com. GitHub issues: https://github.com/saibashirudo/GOSL-MirkFall/issues. Expected response time: within 7 days.
