```dart
import 'package:flutter_local_notifications/flutter_local_notifications.dart';
import 'package:mirkfall/config/constants.dart';
import 'package:mirkfall/domain/ids/session_id.dart';

/// Guarantees the persistent-tracking notification channel exists and can
/// show/dismiss notifications. The ACTIVE-tracking notification itself is
/// emitted by geolocator's ForegroundNotificationConfig (see
/// location_settings_factory.dart); this service owns the channel lifecycle
/// plus the "tap to resume" notification used by Plan 05-06 auto-resume.
class SessionNotificationService {
  SessionNotificationService(this._plugin);
  final FlutterLocalNotificationsPlugin _plugin;

  static const _channel = AndroidNotificationChannel(
    kNotificationChannelId,
    'MirkFall session tracking',
    description: 'Notification persistante pendant une session active',
    importance: Importance.low,
    enableVibration: false,
    playSound: false,
  );

  bool _initialized = false;

  /// Idempotent — creates the Android channel and requests iOS permission.
  Future<void> initialize() async {
    if (_initialized) return;
    final android = _plugin.resolvePlatformSpecificImplementation<
        AndroidFlutterLocalNotificationsPlugin>();
    await android?.createNotificationChannel(_channel);
    final ios = _plugin.resolvePlatformSpecificImplementation<
        IOSFlutterLocalNotificationsPlugin>();
    await ios?.requestPermissions(alert: true, badge: false, sound: false);
    _initialized = true;
  }

  /// Emits a local "tap to resume tracking" notification (Plan 05-06).
  Future<void> showResumeNotification(SessionId id, String displayName) async {
    await initialize();
    await _plugin.show(
      1001,
      'Session "$displayName" interrompue',
      'Tap pour reprendre le suivi',
      NotificationDetails(
        android: AndroidNotificationDetails(
          kNotificationChannelId,
          'MirkFall session tracking',
          importance: Importance.high,
          ongoing: false,
        ),
        iOS: const DarwinNotificationDetails(),
      ),
      payload: 'resume:${id.value}',
    );
  }

  /// Dismisses any MirkFall notification (invoked on Stop).
  Future<void> dismiss() async {
    await _plugin.cancel(1001);
  }
}
```
