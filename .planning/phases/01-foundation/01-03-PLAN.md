`tool/check_headers.dart` — scanne `lib/` ET `test/` ET `tool/` (récursif) pour les fichiers `.dart` non-exclus et vérifie que les 3 premières lignes non-vides sont exactement le header GOSL :

```dart
// Copyright (c) 2026 THONGVAN Alexis
// Licensed under the Good Old Software License v1.0
// See LICENSE file for details

import 'dart:io';

const _expectedHeader = '''// Copyright (c) 2026 THONGVAN Alexis
// Licensed under the Good Old Software License v1.0
// See LICENSE file for details''';

final _excludePatterns = <RegExp>[
  RegExp(r'\.g\.dart$'),
  RegExp(r'\.freezed\.dart$'),
  RegExp(r'\.gr\.dart$'),
  RegExp(r'\.config\.dart$'),
  RegExp(r'[/\\]generated[/\\]'),
  RegExp(r'[/\\]\.dart_tool[/\\]'),
  RegExp(r'[/\\]build[/\\]'),
];

const _defaultRoots = <String>['lib', 'test', 'tool'];

Future<int> main(List<String> args) async {
  final roots = args.isNotEmpty ? args : _defaultRoots;
  final failures = <String>[];
  var scanned = 0;

  for (final rootPath in roots) {
    final root = Directory(rootPath);
    if (!await root.exists()) continue;

    await for (final entity in root.list(recursive: true, followLinks: false)) {
      if (entity is! File) continue;
      final normalized = entity.path.replaceAll('\\', '/');
      if (!normalized.endsWith('.dart')) continue;
      if (_excludePatterns.any((re) => re.hasMatch(normalized))) continue;

      scanned++;
      final contents = await entity.readAsString();
      if (!contents.startsWith(_expectedHeader)) {
        failures.add(entity.path);
      }
    }
  }

  if (failures.isEmpty) {
    stdout.writeln('check_headers: OK ($scanned files)');
    return 0;
  }
  stderr.writeln('check_headers: ${failures.length} file(s) missing GOSL v1.0 header:');
  for (final f in failures) {
    stderr.writeln('  - $f');
  }
  stderr.writeln('\nExpected exact header (3 lines, no trailing blank):');
  stderr.writeln(_expectedHeader);
  exitCode = 1;
  return 1;
}
```

<check_licenses_spec>
`tool/check_licenses.dart` — scanne `pubspec.lock` + `.dart_tool/package_config.json` pour chaque package, lit son `LICENSE` / `pubspec.yaml`, SPDX-match contre l'allowlist :

```dart
// Copyright (c) 2026 THONGVAN Alexis
// Licensed under the Good Old Software License v1.0
// See LICENSE file for details

import 'dart:convert';
import 'dart:io';

import 'package:path/path.dart' as p;
import 'package:yaml/yaml.dart';

const _allowedSpdx = <String>{
  'MIT',
  'BSD-2-Clause',
  'BSD-3-Clause',
  'Apache-2.0',
  'Unlicense',
  'CC0-1.0',
  'ISC',
  'Zlib',
};

/// Known-hard packages: hardcoded SPDX when LICENSE file doesn't auto-parse.
/// Update this map deliberately, commenting each entry with provenance.
const _manualOverrides = <String, String>{
  // Example to document on first failure:
  // 'example_dual_licensed': 'Apache-2.0 OR BSD-3-Clause',
};

/// Forbidden substrings in the LICENSE text — automatic fail.
const _forbiddenSubstrings = <String>[
  'GNU GENERAL PUBLIC LICENSE',
  'GNU LESSER GENERAL PUBLIC LICENSE',
  'GNU AFFERO GENERAL PUBLIC LICENSE',
  'Mozilla Public License',       // MPL is weak copyleft — conservative: flag as non-allowed, human confirms
];

Future<int> main(List<String> args) async {
  final repoRoot = args.isNotEmpty ? args.first : '.';
  final lockPath = p.join(repoRoot, 'pubspec.lock');
  final configPath = p.join(repoRoot, '.dart_tool', 'package_config.json');

  final lockFile = File(lockPath);
  final configFile = File(configPath);
  if (!await lockFile.exists()) {
    stderr.writeln('check_licenses: pubspec.lock not found at $lockPath');
    return 2;
  }
  if (!await configFile.exists()) {
    stderr.writeln('check_licenses: .dart_tool/package_config.json not found. Run `flutter pub get` first.');
    return 2;
  }

  final lock = loadYaml(await lockFile.readAsString()) as YamlMap;
  final packages = (lock['packages'] as YamlMap?) ?? YamlMap();

  final configJson = jsonDecode(await configFile.readAsString()) as Map<String, Object?>;
  final rootUriByPackage = <String, String>{};
  for (final pkg in (configJson['packages'] as List).cast<Map<String, Object?>>()) {
    rootUriByPackage[pkg['name'] as String] = pkg['rootUri'] as String;
  }

  final violations = <String>[];
  final unresolved = <String>[];
  var ok = 0;

  for (final entry in packages.entries) {
    final name = entry.key as String;
    final meta = entry.value as YamlMap;
    final source = meta['source'] as String? ?? 'unknown';
    if (source == 'sdk') continue; // Flutter SDK itself, BSD-3-Clause, skip.

    final spdx = _manualOverrides[name] ?? await _resolveSpdx(name, rootUriByPackage[name], repoRoot);

    if (spdx == null) {
      unresolved.add(name);
      continue;
    }

    // Split compound expressions: "Apache-2.0 OR BSD-3-Clause" passes if either is allowed.
    final ids = spdx.split(RegExp(r'\s+OR\s+')).map((s) => s.trim()).toList();
    final allowed = ids.any(_allowedSpdx.contains);
    if (!allowed) {
      violations.add('$name: $spdx NOT in allowlist');
    } else {
      ok++;
    }
  }

  if (violations.isEmpty && unresolved.isEmpty) {
    stdout.writeln('check_licenses: OK ($ok packages)');
    return 0;
  }
  if (violations.isNotEmpty) {
    stderr.writeln('check_licenses: ${violations.length} violation(s):');
    for (final v in violations) {
      stderr.writeln('  - $v');
    }
  }
  if (unresolved.isNotEmpty) {
    stderr.writeln('check_licenses: ${unresolved.length} package(s) could not be resolved. Add to _manualOverrides with source link:');
    for (final n in unresolved) {
      stderr.writeln('  - $n');
    }
  }
  exitCode = 1;
  return 1;
}

/// Attempt to resolve a package's SPDX identifier by:
/// 1) reading its pubspec.yaml's `license:` field if present,
/// 2) reading a LICENSE / LICENSE.md / LICENSE.txt at the package root and fuzzy-matching.
Future<String?> _resolveSpdx(String name, String? rootUri, String repoRoot) async {
  if (rootUri == null) return null;
  // rootUri comes as "file:///..." OR "../../..". Resolve against .dart_tool/.
  final configDir = p.join(repoRoot, '.dart_tool');
  final packageDir = Uri.parse(rootUri).scheme == 'file'
      ? Uri.parse(rootUri).toFilePath()
      : p.normalize(p.join(configDir, Uri.parse(rootUri).path));

  // 1. Check pubspec.yaml `license:` field (rare but authoritative).
  final pubspecFile = File(p.join(packageDir, 'pubspec.yaml'));
  if (await pubspecFile.exists()) {
    final pubspec = loadYaml(await pubspecFile.readAsString()) as YamlMap?;
    final declared = pubspec?['license'] as String?;
    if (declared != null && declared.isNotEmpty) return declared.trim();
  }

  // 2. Read LICENSE / LICENSE.md / LICENSE.txt
  for (final candidate in <String>['LICENSE', 'LICENSE.md', 'LICENSE.txt']) {
    final f = File(p.join(packageDir, candidate));
    if (await f.exists()) {
      final text = await f.readAsString();
      // Forbidden substrings → bail immediately as "unknown/forbidden".
      for (final bad in _forbiddenSubstrings) {
        if (text.contains(bad)) {
          return 'UNKNOWN-FORBIDDEN-MARKER: $bad';
        }
      }
      // Heuristic SPDX match.
      if (text.contains('Apache License') && text.contains('Version 2.0')) return 'Apache-2.0';
      if (text.contains('MIT License') || text.contains('Permission is hereby granted, free of charge,')) {
        // MIT has a signature phrase. But BSD-3-Clause is close; disambiguate on "Redistributions of source code".
        if (text.contains('Redistributions of source code must retain')) {
          // BSD family
          if (text.contains('Neither the name of')) return 'BSD-3-Clause';
          return 'BSD-2-Clause';
        }
        return 'MIT';
      }
      if (text.contains('Redistributions of source code must retain')) {
        if (text.contains('Neither the name of')) return 'BSD-3-Clause';
        return 'BSD-2-Clause';
      }
      if (text.contains('ISC License') || text.contains('ISC license')) return 'ISC';
      if (text.contains('Unlicense')) return 'Unlicense';
      if (text.contains('CC0 1.0 Universal')) return 'CC0-1.0';
      if (text.contains('zlib License')) return 'Zlib';
    }
  }

  return null;
}
```

**NOTE workaround:** le pattern `UNKNOWN-FORBIDDEN-MARKER: GNU GENERAL PUBLIC LICENSE` retourné comme SPDX ne sera PAS dans l'allowlist, donc il fera échouer le check — comportement voulu. Phase 15 pourra durcir le parsing si besoin.

**NOTE strict-casts:** `YamlMap` est un `Map<Object?, Object?>`, les casts comme `meta['source'] as String?` sont nécessaires et le comportement est correct.

<check_deps_md_spec>
`tool/check_dependencies_md.dart` — parse `pubspec.lock` + `DEPENDENCIES.md` (markdown tables), cross-référence, erreurs diff-style :

```dart
// Copyright (c) 2026 THONGVAN Alexis
// Licensed under the Good Old Software License v1.0
// See LICENSE file for details

import 'dart:io';

import 'package:path/path.dart' as p;
import 'package:yaml/yaml.dart';

/// Cross-reference between `pubspec.lock` and `DEPENDENCIES.md`.
/// Every package in the lock (non-SDK) MUST appear in DEPENDENCIES.md with the
/// same version. The markdown tables are parsed by looking for `| name | version | ...`
/// rows in either the "Direct dependencies" or "Transitive dependencies" sections.
Future<int> main(List<String> args) async {
  final repoRoot = args.isNotEmpty ? args.first : '.';
  final lockPath = p.join(repoRoot, 'pubspec.lock');
  final depsMdPath = p.join(repoRoot, 'DEPENDENCIES.md');

  final lockFile = File(lockPath);
  final mdFile = File(depsMdPath);
  if (!await lockFile.exists()) {
    stderr.writeln('check_dependencies_md: pubspec.lock not found at $lockPath');
    return 2;
  }
  if (!await mdFile.exists()) {
    stderr.writeln('check_dependencies_md: DEPENDENCIES.md not found at $depsMdPath');
    return 2;
  }

  final lock = loadYaml(await lockFile.readAsString()) as YamlMap;
  final lockPackages = (lock['packages'] as YamlMap?) ?? YamlMap();

  // Build the "expected" map: {name -> version} from pubspec.lock (skip SDK entries).
  final expected = <String, String>{};
  for (final entry in lockPackages.entries) {
    final name = entry.key as String;
    final meta = entry.value as YamlMap;
    final source = meta['source'] as String? ?? 'unknown';
    if (source == 'sdk') continue;
    final version = meta['version'] as String? ?? '';
    expected[name] = version;
  }

  // Parse DEPENDENCIES.md tables: any line starting with `| ` with at least 6 pipe-separated cells.
  final mdText = await mdFile.readAsString();
  final declared = <String, String>{};
  for (final line in mdText.split('\n')) {
    if (!line.startsWith('| ')) continue;
    final cells = line.split('|').map((c) => c.trim()).toList();
    // Expect: empty, name, version, license, source, telemetry_audit, date, empty
    if (cells.length < 7) continue;
    final name = cells[1];
    final version = cells[2];
    // Filter out header + separator rows.
    if (name.isEmpty || name == 'Package' || name == '---' || name.startsWith('---')) continue;
    if (version.isEmpty || version == 'Version' || version.startsWith('-')) continue;
    declared[name] = version;
  }

  final missing = <String>[];
  final extra = <String>[];
  final mismatched = <String>[];

  for (final e in expected.entries) {
    if (!declared.containsKey(e.key)) {
      missing.add('${e.key} ${e.value}');
    } else if (declared[e.key] != e.value) {
      mismatched.add('${e.key}: lock=${e.value} md=${declared[e.key]}');
    }
  }
  for (final d in declared.keys) {
    if (!expected.containsKey(d)) {
      extra.add(d);
    }
  }

  if (missing.isEmpty && extra.isEmpty && mismatched.isEmpty) {
    stdout.writeln('check_dependencies_md: OK (${expected.length} packages)');
    return 0;
  }

  if (missing.isNotEmpty) {
    stderr.writeln('check_dependencies_md: ${missing.length} package(s) in pubspec.lock MISSING from DEPENDENCIES.md:');
    for (final m in missing) {
      stderr.writeln('  - $m');
    }
  }
  if (extra.isNotEmpty) {
    stderr.writeln('check_dependencies_md: ${extra.length} package(s) in DEPENDENCIES.md NOT in pubspec.lock:');
    for (final m in extra) {
      stderr.writeln('  - $m');
    }
  }
  if (mismatched.isNotEmpty) {
    stderr.writeln('check_dependencies_md: ${mismatched.length} version mismatch(es):');
    for (final m in mismatched) {
      stderr.writeln('  - $m');
    }
  }
  exitCode = 1;
  return 1;
}
```
