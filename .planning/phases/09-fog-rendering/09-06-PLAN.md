```dart
import 'dart:async';
import 'dart:developer' show log;

import '../../config/constants.dart';
import '../../domain/revealed/reveal_calculator.dart';
import '../../domain/revealed/revealed_tile_store.dart';
import '../../domain/revealed/tile_math.dart';
import '../../domain/sessions/session_id.dart';
import '../../domain/tracking/fix.dart';

/// Buffers GPS fixes and flushes reveal-mask writes to [RevealedTileStore]
/// in batches.
///
/// Flush triggers (first-to-fire wins):
/// * [kRevealFlushIntervalSeconds] elapsed since the oldest buffered fix
/// * [kRevealFlushMaxFixes] fixes buffered
///
/// Per-fix reveal radius: [kDefaultRevealRadiusMeters] (25 m). Initial
/// session-open reveal uses [kInitialRevealRadiusMeters] (20 m) via
/// [revealInitial].
///
/// Disposal semantics: [dispose] awaits an in-flight flush, then writes
/// any still-buffered fixes synchronously before returning — guarantees
/// no fix is lost on session end / app background.
class RevealStreamingController {
  RevealStreamingController({
    required this.sessionId,
    required this.store,
    this.flushInterval = const Duration(seconds: kRevealFlushIntervalSeconds),
    this.flushMaxFixes = kRevealFlushMaxFixes,
    this.revealRadiusMeters = kDefaultRevealRadiusMeters,
  });

  final SessionId sessionId;
  final RevealedTileStore store;
  final Duration flushInterval;
  final int flushMaxFixes;
  final double revealRadiusMeters;

  final List<Fix> _buffer = [];
  Timer? _flushTimer;
  bool _disposed = false;
  Future<void>? _inFlightFlush;

  /// Called by [ActiveSessionController] whenever a new Fix arrives.
  Future<void> onFix(Fix fix) async {
    if (_disposed) return;
    _buffer.add(fix);
    if (_buffer.length >= flushMaxFixes) {
      await _flush();
      return;
    }
    _flushTimer ??= Timer(flushInterval, () { _flush(); });
  }

  /// Writes an initial N-metre reveal around [fix] (MIRK session-open UX).
  /// Not buffered — writes synchronously (well, awaited) so the UI sees
  /// the reveal immediately.
  Future<void> revealInitial(Fix fix) async {
    if (_disposed) return;
    await _writeCircleReveal(fix.lat, fix.lon, kInitialRevealRadiusMeters);
  }

  /// Force an immediate flush. Idempotent — multiple calls await the
  /// single in-flight future.
  Future<void> flush() => _flush();

  Future<void> _flush() async {
    // re-entrancy guard: coalesce overlapping flush invocations
    if (_inFlightFlush != null) return _inFlightFlush!;
    _flushTimer?.cancel();
    _flushTimer = null;
    if (_buffer.isEmpty) return;

    final batch = List<Fix>.from(_buffer);
    _buffer.clear();

    _inFlightFlush = _flushBatch(batch);
    try {
      await _inFlightFlush;
    } finally {
      _inFlightFlush = null;
    }
  }

  Future<void> _flushBatch(List<Fix> batch) async {
    for (final fix in batch) {
      await _writeCircleReveal(fix.lat, fix.lon, revealRadiusMeters);
    }
  }

  Future<void> _writeCircleReveal(double lat, double lon, double radius) async {
    final touchedTiles = _touchedParentTiles(lat, lon, radius);
    for (final tile in touchedTiles) {
      final mask = computeRevealMask(
        centerLat: lat, centerLon: lon, radiusMeters: radius,
        parentX: tile.x, parentY: tile.y, parentZoom: kRevealedTileParentZoom,
      );
      // only write if any bit is set — saves DB round-trips on edge cases
      if (!_hasAnyBit(mask)) continue;
      try {
        await store.mergeMask(
          sessionId: sessionId,
          parentX: tile.x, parentY: tile.y,
          mask: mask,
        );
      } catch (e, stack) {
        log('RevealStreamingController: mergeMask failed',
            error: e, stackTrace: stack,
            name: 'application.reveal');
        // Do not rethrow — a single tile write failure should not drop
        // the entire batch. Fix is lost but logged; user can re-walk the
        // area. CLAUDE.md §Error handling level 2 (external error, log + continue).
      }
    }
  }

  /// Set of parent tiles at z=14 touched by a circle centred at (lat, lon)
  /// with [radius] meters. Typically 1; up to 4 in worst case.
  Iterable<({int x, int y})> _touchedParentTiles(
      double lat, double lon, double radius) {
    // Compute circle bbox (same math as computeRevealMask helper)
    const latDegPerMetre = 1.0 / 111_320.0;
    final lonDegPerMetre = 1.0 /
        (111_320.0 * _cos(lat * _toRad));
    final minLat = lat - radius * latDegPerMetre;
    final maxLat = lat + radius * latDegPerMetre;
    final minLon = lon - radius * lonDegPerMetre;
    final maxLon = lon + radius * lonDegPerMetre;
    // Convert bbox corners to tile coords
    final nw = TileMath.latLonToTile(
        lat: maxLat, lon: minLon, zoom: kRevealedTileParentZoom);
    final se = TileMath.latLonToTile(
        lat: minLat, lon: maxLon, zoom: kRevealedTileParentZoom);
    return [
      for (var y = nw.y; y <= se.y; y++)
        for (var x = nw.x; x <= se.x; x++) (x: x, y: y)
    ];
  }

  static const double _toRad = 3.141592653589793 / 180.0;
  static double _cos(double x) => _mathCos(x);
  // Import dart:math's cos via local helper to avoid top-level `import 'dart:math' as math`
  // noise in the controller. Keep this file math-helper-terse.
}

// Top-level helper (could be private via file-local)
double _mathCos(double x) => math.cos(x);
```

```dart
/// Session-scoped reveal streaming controller.
///
/// Returns null when no session is active (the UI hides the fog overlay
/// in that case). When a session is active, builds a single
/// [RevealStreamingController] instance; `ref.onDispose` flushes and
/// disposes it when the session ends.
@riverpod
RevealStreamingController? revealStreamingController(Ref ref) {
  final sessionState = ref.watch(activeSessionControllerProvider);
  final sessionId = sessionState.valueOrNull?.whenOrNull(
    tracking: (session, _) => session.id,
  );
  if (sessionId == null) return null;

  final store = ref.watch(revealedTileStoreProvider);
  final controller = RevealStreamingController(
    sessionId: sessionId,
    store: store,
  );
  ref.onDispose(() async {
    // Note: onDispose is sync; we schedule a microtask for the async flush
    // and rely on the app not being torn down mid-flush. Acceptable in
    // practice — Phase 05 FixStore.flush has the same pattern.
    await controller.flush();
    // dispose doesn't currently await, so no return value to capture
  });
  return controller;
}
```

```dart
import 'dart:developer' show log;

import '../../domain/mirk/mirk_style_id.dart';
import '../../domain/mirk/mirk_style_store.dart';
import '../../domain/sessions/session_id.dart';
import '../../domain/sessions/session_store.dart';

class MirkStyleSessionController {
  MirkStyleSessionController({
    required this.sessionStore,
    required this.styleStore,
    required this.invalidateRenderer,  // typed Function
  });
  final SessionStore sessionStore;
  final MirkStyleStore styleStore;
  final void Function() invalidateRenderer;

  /// Sets [styleId] as the active style for [sessionId], persists to DB,
  /// and invalidates the active-renderer provider so the next frame
  /// picks up the new renderer.
  ///
  /// Throws [NoActiveSessionException] if [sessionId] does not resolve
  /// to a row in t_sessions. Throws [MirkStyleNotFoundException] if
  /// [styleId] does not resolve to a row in t_mirk_styles (defensive;
  /// prevents FK violation).
  Future<void> select({
    required SessionId sessionId,
    required MirkStyleId styleId,
  }) async {
    // Defensive: confirm the style exists before writing the FK.
    final style = await styleStore.findById(styleId);
    if (style == null) {
      throw MirkStyleNotFoundException(styleId);
    }
    final session = await sessionStore.findById(sessionId);
    if (session == null) {
      throw NoActiveSessionException(sessionId);
    }
    if (session.mirkStyleId == styleId) {
      // already selected — no-op
      log('MirkStyleSessionController: styleId already active, skipping',
          name: 'application.mirk.style');
      return;
    }
    await sessionStore.updateMirkStyle(sessionId: sessionId, mirkStyleId: styleId);
    invalidateRenderer();
  }
}

class MirkStyleNotFoundException implements Exception {
  MirkStyleNotFoundException(this.styleId);
  final MirkStyleId styleId;
  @override
  String toString() => 'MirkStyleNotFoundException(styleId=\$styleId)';
}

class NoActiveSessionException implements Exception {
  NoActiveSessionException(this.sessionId);
  final SessionId sessionId;
  @override
  String toString() => 'NoActiveSessionException(sessionId=\$sessionId)';
}
```
