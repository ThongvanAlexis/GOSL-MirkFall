```dart
import 'dart:async';
import 'dart:developer' show log;
import 'dart:math' as math;
import 'dart:typed_data';

import '../../config/constants.dart';
import '../../domain/fixes/fix.dart';
import '../../domain/ids/session_id.dart';
import '../../domain/revealed/reveal_calculator.dart';
import '../../domain/revealed/revealed_tile_store.dart';
import '../../domain/revealed/tile_math.dart';

/// Buffers GPS fixes and flushes reveal-mask writes to [RevealedTileStore]
/// in batches.
///
/// Flush triggers (first-to-fire wins):
/// * [kRevealFlushIntervalSeconds] elapsed since the oldest buffered fix
/// * [kRevealFlushMaxFixes] fixes buffered
///
/// Per-fix reveal radius: [kDefaultRevealRadiusMeters] (25 m). Initial
/// session-open reveal uses [kInitialRevealRadiusMeters] (20 m) via
/// [revealInitial].
///
/// Disposal semantics: [dispose] flushes any still-buffered fixes
/// synchronously before returning — guarantees no fix is lost on session
/// end / app background.
class RevealStreamingController {
  RevealStreamingController({
    required this.sessionId,
    required this.store,
    this.flushInterval = const Duration(seconds: kRevealFlushIntervalSeconds),
    this.flushMaxFixes = kRevealFlushMaxFixes,
    this.revealRadiusMeters = kDefaultRevealRadiusMeters,
  });

  final SessionId sessionId;
  final RevealedTileStore store;
  final Duration flushInterval;
  final int flushMaxFixes;
  final double revealRadiusMeters;

  final List<Fix> _buffer = [];
  Timer? _flushTimer;
  bool _disposed = false;
  Future<void>? _inFlightFlush;

  Future<void> onFix(Fix fix) async {
    if (_disposed) return;
    _buffer.add(fix);
    if (_buffer.length >= flushMaxFixes) {
      await _flush();
      return;
    }
    _flushTimer ??= Timer(flushInterval, _flush);
  }

  Future<void> revealInitial(Fix fix) async {
    if (_disposed) return;
    await _writeCircleReveal(fix.lat, fix.lon, kInitialRevealRadiusMeters);
  }

  Future<void> flush() => _flush();

  Future<void> _flush() async {
    if (_inFlightFlush != null) return _inFlightFlush!;
    _flushTimer?.cancel();
    _flushTimer = null;
    if (_buffer.isEmpty) return;

    final batch = List<Fix>.from(_buffer);
    _buffer.clear();

    _inFlightFlush = _flushBatch(batch);
    try {
      await _inFlightFlush;
    } finally {
      _inFlightFlush = null;
    }
  }

  Future<void> _flushBatch(List<Fix> batch) async {
    for (final fix in batch) {
      await _writeCircleReveal(fix.lat, fix.lon, revealRadiusMeters);
    }
  }

  Future<void> _writeCircleReveal(double lat, double lon, double radius) async {
    final touchedTiles = _touchedParentTiles(lat, lon, radius);
    for (final tile in touchedTiles) {
      final mask = computeRevealMask(
        centerLat: lat, centerLon: lon, radiusMeters: radius,
        parentX: tile.x, parentY: tile.y, parentZoom: kRevealedTileParentZoom,
      );
      if (!_hasAnyBit(mask)) continue;
      try {
        await store.mergeMask(
          sessionId: sessionId,
          parentX: tile.x, parentY: tile.y,
          mask: mask,
        );
      } catch (e, stack) {
        log('RevealStreamingController: mergeMask failed',
            error: e, stackTrace: stack, name: 'application.reveal');
        // Do not rethrow — single-tile write failure should not drop the
        // whole batch. CLAUDE.md §Error handling level 2.
      }
    }
  }

  Iterable<({int x, int y})> _touchedParentTiles(double lat, double lon, double radius) {
    const latDegPerMetre = 1.0 / 111_320.0;
    final lonDegPerMetre = 1.0 / (111_320.0 * math.cos(lat * math.pi / 180.0));
    final minLat = lat - radius * latDegPerMetre;
    final maxLat = lat + radius * latDegPerMetre;
    final minLon = lon - radius * lonDegPerMetre;
    final maxLon = lon + radius * lonDegPerMetre;
    final nw = TileMath.latLonToTile(lat: maxLat, lon: minLon, zoom: kRevealedTileParentZoom);
    final se = TileMath.latLonToTile(lat: minLat, lon: maxLon, zoom: kRevealedTileParentZoom);
    return [
      for (var y = nw.y; y <= se.y; y++)
        for (var x = nw.x; x <= se.x; x++) (x: x, y: y)
    ];
  }

  bool _hasAnyBit(Uint8List bytes) {
    for (final b in bytes) { if (b != 0) return true; }
    return false;
  }

  Future<void> dispose() async {
    _disposed = true;
    _flushTimer?.cancel();
    await _flush();  // flush any remaining buffered fixes
  }
}
```
