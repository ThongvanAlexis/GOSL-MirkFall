/// Freezed view of a lat/lon bbox decoupled from MapLibre types.
///
/// Represented as four doubles â€” NOT `LatLngBounds` â€” so consumers in
/// `lib/domain/` and the `MirkPaintContext` stay MapLibre-type-free per
/// the MAP-06 seam discipline. A thin adapter in
/// `lib/infrastructure/map/` converts `LatLngBounds` â†’ `MirkViewportBbox`
/// at the platform boundary.
///
/// ## Antimeridian wrap
///
/// `east < west` is permitted when the viewport crosses the Â±180Â° line
/// (west > 0 && east < 0). The primary user of this semantic is the
/// Phase 09 `RevealStreamingController.visibleParentTilesAtZ14` helper.
@freezed
abstract class MirkViewportBbox with _$MirkViewportBbox {
  @Assert('south <= north',
      'MirkViewportBbox: south must be <= north (got south=\$south, north=\$north)')
  @Assert('west <= east || (west > 0 && east < 0)',
      'MirkViewportBbox: east < west only permitted on antimeridian wrap')

    required double south,
    required double west,
    required double north,
