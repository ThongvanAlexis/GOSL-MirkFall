**Source of truth**: `tool/fixtures/build_50k_tiles.dart` deterministic builder.
**Committed artefact**: `test/fixtures/mirk/fifty_k_tiles_seed.sql` — text SQL with 50,000 INSERT statements.
**Integrity gate**: `tool/check_mirk_fixture_fresh.dart` runs the builder to a tmp file and byte-compares.

**Builder parameters** (hardcoded constants in the builder file):
- Session: `id = 'sess_01FIFTYKTEST0000000000000'` (ULID-shaped deterministic ID)
- Tile grid: 500 × 100 starting at (parentX=8400, parentY=5500), parentZoom=14 — Europe mid-latitude (matches plan 09-03 test latitudes)
- Bitmap density: ~25% bits set (1024 of 4096), distribution seeded by `math.Random(kFiftyKSeed)` where `kFiftyKSeed = 0x4D49524B` (ASCII "MIRK" = 0x4D 49 52 4B)
- `set_bit_count`: exact popcount of each bitmap (consistency with Phase 03 MIRK-03 audit trail)
- `updated_at_utc`: fixed ISO string `'2026-01-01T00:00:00.000Z'` — deterministic

**Size budget**: 50,000 × ~200 bytes/row (hex-encoded BLOB + INSERT prefix) ≈ 10 MB text. Commit to git (no LFS). If size balloons beyond 50 MB, switch to per-row BLOB encoding or commit a compressed `.sql.gz` with decompression in the perf test.

**Hex-encoded bitmap**: Drift supports `X'...'` hex literal syntax. Each INSERT:


```dart
const int kFiftyKSeed = 0x4D49524B;  // "MIRK" in ASCII
const int kFiftyKRowCount = 50000;
const String kFiftyKSessionId = 'sess_01FIFTYKTEST0000000000000';
const int kFiftyKGridWidth = 500;
const int kFiftyKGridHeight = 100;
const int kFiftyKGridOriginX = 8400;
const int kFiftyKGridOriginY = 5500;
const int kFiftyKParentZoom = 14;
const String kFiftyKCreatedAtUtc = '2026-01-01T00:00:00.000Z';
const double kFiftyKBitDensity = 0.25;

void main() {
  final outputPath = 'test/fixtures/mirk/fifty_k_tiles_seed.sql';
  final buffer = StringBuffer();
  buffer.writeln('-- Phase 09 plan 09-08 — deterministic 50k t_revealed_tiles fixture.');
  buffer.writeln('-- Seed: 0x4D49524B ("MIRK"). Builder: tool/fixtures/build_50k_tiles.dart.');
  buffer.writeln('-- DO NOT HAND-EDIT — CI gate check_mirk_fixture_fresh.dart enforces freshness.');
  buffer.writeln();
  buffer.writeln('-- Session row (prerequisite for FK on t_revealed_tiles.session_id)');
  buffer.writeln("INSERT OR IGNORE INTO t_sessions (id, display_name, status, started_at_utc, started_at_offset_minutes, stopped_at_utc, stopped_at_offset_minutes, mirk_style_id) VALUES ('\$kFiftyKSessionId', '50k Perf Fixture', 'stopped', '\$kFiftyKCreatedAtUtc', 0, '\$kFiftyKCreatedAtUtc', 0, NULL);");
  buffer.writeln();
  buffer.writeln('-- 50,000 revealed-tile rows');

  final rng = math.Random(kFiftyKSeed);
  var rowCounter = 0;
  for (var gy = 0; gy < kFiftyKGridHeight; gy++) {
    for (var gx = 0; gx < kFiftyKGridWidth; gx++) {
      final parentX = kFiftyKGridOriginX + gx;
      final parentY = kFiftyKGridOriginY + gy;
      final bitmap = _generateBitmap(rng);
      final setBitCount = _popcount(bitmap);
      final hex = _toHex(bitmap);
      final rowId = 'rtil_01FIFTY\${rowCounter.toString().padLeft(5, '0')}';
      buffer.writeln("INSERT INTO t_revealed_tiles (id, session_id, parent_x, parent_y, parent_zoom, bitmap, set_bit_count, updated_at_utc) VALUES ('\$rowId', '\$kFiftyKSessionId', \$parentX, \$parentY, \$kFiftyKParentZoom, X'\$hex', \$setBitCount, '\$kFiftyKCreatedAtUtc');");
      rowCounter++;
    }
  }

  File(outputPath).writeAsStringSync(buffer.toString());
  stdout.writeln('Wrote \$rowCounter rows to \$outputPath');
}

Uint8List _generateBitmap(math.Random rng) {
  final bytes = Uint8List(512);
  for (var i = 0; i < 4096; i++) {
    if (rng.nextDouble() < kFiftyKBitDensity) {
      bytes[i >> 3] |= 1 << (i & 7);
    }
  }
  return bytes;
}

int _popcount(Uint8List bytes) {
  var count = 0;
  for (final b in bytes) {
    var v = b;
    v = v - ((v >> 1) & 0x55);
    v = (v & 0x33) + ((v >> 2) & 0x33);
    count += (v + (v >> 4)) & 0x0F;
  }
  return count;
}

String _toHex(Uint8List bytes) =>
    bytes.map((b) => b.toRadixString(16).padLeft(2, '0').toUpperCase()).join();
```

```dart
import 'dart:io';

const String kFixturePath = 'test/fixtures/mirk/fifty_k_tiles_seed.sql';
const String kBuilderPath = 'tool/fixtures/build_50k_tiles.dart';

Future<int> main(List<String> args) async {
  final committed = File(kFixturePath);
  if (!committed.existsSync()) {
    stderr.writeln('check_mirk_fixture_fresh: FAIL — committed fixture missing at \$kFixturePath');
    stderr.writeln('  Run: dart run \$kBuilderPath');
    return 1;
  }

  final tmpDir = await Directory.systemTemp.createTemp('mirk_fixture_');
  try {
    // Copy builder to tmp AND point it at a tmp output
    // Simplest: run builder with env var or argv override for outputPath
    // Or: just run it (writes to the committed path) then diff with stash.
    //
    // Cleanest: the builder should accept `--output=...` argv.
    // Extend build_50k_tiles.dart to honour --output argv, default to
    // kFixturePath. Here we pass --output=<tmp file>.

    final tmpOutput = '\${tmpDir.path}/fifty_k_tiles_seed.sql';
    final proc = await Process.run(
      'dart',
      ['run', kBuilderPath, '--output=\$tmpOutput'],
    );
    if (proc.exitCode != 0) {
      stderr.writeln('check_mirk_fixture_fresh: FAIL — builder exited \${proc.exitCode}');
      stderr.writeln(proc.stderr);
      return 2;
    }

    final committedBytes = committed.readAsBytesSync();
    final freshBytes = File(tmpOutput).readAsBytesSync();

    if (committedBytes.length != freshBytes.length) {
      stderr.writeln('check_mirk_fixture_fresh: FAIL — size mismatch');
      stderr.writeln('  Committed: \${committedBytes.length} bytes');
      stderr.writeln('  Fresh:     \${freshBytes.length} bytes');
      stderr.writeln('  Fix: dart run \$kBuilderPath');
      return 1;
    }
    for (var i = 0; i < committedBytes.length; i++) {
      if (committedBytes[i] != freshBytes[i]) {
        stderr.writeln('check_mirk_fixture_fresh: FAIL — byte \$i differs');
        stderr.writeln('  Fix: dart run \$kBuilderPath');
        return 1;
      }
    }
    stdout.writeln('check_mirk_fixture_fresh: OK — \${committedBytes.length} bytes match');
    return 0;
  } finally {
    tmpDir.deleteSync(recursive: true);
  }
}
```
