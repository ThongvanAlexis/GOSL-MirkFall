<algorithm>
From 09-RESEARCH §computeRevealMask Algorithm Specification — bbox-first + per-cell Haversine:

```
1. Compute parent tile bbox (NW, SE) via TileMath.tileToLatLon.
2. Compute circle bbox: expand (centerLat, centerLon) by ±radiusMeters
   (crude Mercator: 1° lat ≈ 111_320 m; 1° lon ≈ 111_320 * cos(lat) m).
3. If circle bbox does NOT intersect parent tile bbox → return all-zero mask.
4. Compute cell spans + clipped cell index range.
5. For each (i, j) in the range:
   a. Compute cell rectangle corners.
   b. Closest point on cell to circle centre (axis-clamp).
   c. Haversine distance from circle centre to closest point.
   d. If distance ≤ radiusMeters → flip bit (j * 64 + i).
6. Return the 512-byte mask.
```

Helpers (private, top-of-file): `_circleBbox`, `_bboxesIntersect`, `_haversineMeters`, `_clamp`, `_toRad`.

Allowed: `dart:math` (log, sin, cos, atan2, sqrt, pi). NOT allowed: Flutter, Drift, infrastructure layer imports.
</algorithm>

```dart
import 'dart:math' as math;
import 'dart:typed_data';

import 'revealed_tile.dart';  // kRevealedTileBitmapBytes + kRevealedTileBitmapCellsPerSide
import 'tile_math.dart';

Uint8List computeRevealMask({
  required double centerLat,
  required double centerLon,
  required double radiusMeters,
  required int parentX,
  required int parentY,
  required int parentZoom,
}) {
  final mask = Uint8List(kRevealedTileBitmapBytes);
  if (radiusMeters <= 0.0) return mask;  // defensive — no-op

  final parentNw = TileMath.tileToLatLon(x: parentX, y: parentY, zoom: parentZoom);
  final parentSe = TileMath.tileToLatLon(x: parentX + 1, y: parentY + 1, zoom: parentZoom);

  final latDegPerMetre = 1.0 / _metresPerDegreeLat;
  final lonDegPerMetre = 1.0 / (_metresPerDegreeLat * math.cos(_toRad(centerLat)));
  final circleMinLat = centerLat - radiusMeters * latDegPerMetre;
  final circleMaxLat = centerLat + radiusMeters * latDegPerMetre;
  final circleMinLon = centerLon - radiusMeters * lonDegPerMetre;
  final circleMaxLon = centerLon + radiusMeters * lonDegPerMetre;

  // Early exit: circle bbox outside parent bbox
  if (circleMaxLat < parentSe.lat || circleMinLat > parentNw.lat ||
      circleMaxLon < parentNw.lon || circleMinLon > parentSe.lon) {
    return mask;
  }

  const cellsPerSide = kRevealedTileBitmapCellsPerSide;  // 64
  final cellLatSpan = (parentNw.lat - parentSe.lat) / cellsPerSide;
  final cellLonSpan = (parentSe.lon - parentNw.lon) / cellsPerSide;

  final cellJStart = math.max(0, ((parentNw.lat - circleMaxLat) / cellLatSpan).floor());
  final cellJEnd = math.min(cellsPerSide - 1, ((parentNw.lat - circleMinLat) / cellLatSpan).ceil());
  final cellIStart = math.max(0, ((circleMinLon - parentNw.lon) / cellLonSpan).floor());
  final cellIEnd = math.min(cellsPerSide - 1, ((circleMaxLon - parentNw.lon) / cellLonSpan).ceil());

  for (var j = cellJStart; j <= cellJEnd; j++) {
    final cellNorthLat = parentNw.lat - j * cellLatSpan;
    final cellSouthLat = parentNw.lat - (j + 1) * cellLatSpan;
    for (var i = cellIStart; i <= cellIEnd; i++) {
      final cellWestLon = parentNw.lon + i * cellLonSpan;
      final cellEastLon = parentNw.lon + (i + 1) * cellLonSpan;

      // Closest point on cell rectangle to circle centre (axis-clamp)
      final closestLat = _clamp(centerLat, cellSouthLat, cellNorthLat);
      final closestLon = _clamp(centerLon, cellWestLon, cellEastLon);

      final d = _haversineMeters(centerLat, centerLon, closestLat, closestLon);
      if (d <= radiusMeters) {
        final bitIndex = j * cellsPerSide + i;
        final byteIndex = bitIndex >> 3;
        final bitOffset = bitIndex & 7;
        mask[byteIndex] |= 1 << bitOffset;
      }
    }
  }
  return mask;
}

// ---- Helpers (private, top-level) ----

/// Approximate metres per degree of latitude. Good to 0.5 % globally.
/// Per WGS-84 / 2π×R/360 at equator.
const double _metresPerDegreeLat = 111_320.0;

/// WGS-84 mean Earth radius (metres) — used by [_haversineMeters].
const double _earthRadiusMeters = 6_371_008.8;

double _toRad(double deg) => deg * math.pi / 180.0;

double _clamp(double value, double low, double high) {
  if (value < low) return low;
  if (value > high) return high;
  return value;
}

/// Great-circle distance in metres via the Haversine formula.
double _haversineMeters(double lat1, double lon1, double lat2, double lon2) {
  final dLat = _toRad(lat2 - lat1);
  final dLon = _toRad(lon2 - lon1);
  final l1 = _toRad(lat1);
  final l2 = _toRad(lat2);
  final a = math.sin(dLat / 2) * math.sin(dLat / 2) +
            math.sin(dLon / 2) * math.sin(dLon / 2) * math.cos(l1) * math.cos(l2);
  final c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a));
  return _earthRadiusMeters * c;
}
```
