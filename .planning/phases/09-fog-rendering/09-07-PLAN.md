<visible_tiles_provider>
The `visibleMirkTilesProvider` is the viewport filtering seam:

```dart
@riverpod
Future<List<VisibleMirkTile>> visibleMirkTiles(Ref ref) async {
  // Watch active session + viewport
  final sessionState = ref.watch(activeSessionControllerProvider);
  final sessionId = sessionState.valueOrNull?.whenOrNull(
    tracking: (s, _) => s.id,
  );
  if (sessionId == null) return const [];

  // Watch viewport — assume MapView emits MirkViewportBbox on change
  final viewport = ref.watch(mapViewportProvider);  // new or existing
  if (viewport == null) return const [];

  // Compute parent tiles intersecting the viewport at z=14
  final nw = TileMath.latLonToTile(
      lat: viewport.north, lon: viewport.west, zoom: kRevealedTileParentZoom);
  final se = TileMath.latLonToTile(
      lat: viewport.south, lon: viewport.east, zoom: kRevealedTileParentZoom);

  final store = ref.watch(revealedTileStoreProvider);
  final result = <VisibleMirkTile>[];
  for (var y = nw.y; y <= se.y; y++) {
    for (var x = nw.x; x <= se.x; x++) {
      final row = await store.findByParent(
          sessionId: sessionId, parentX: x, parentY: y);
      // Null = no bits revealed yet for this tile. Still include it — all-zero bitmap
      // means the entire tile is fog.
      final bitmap = row?.bitmap ?? Uint8List(kRevealedTileBitmapBytes);
      final nwLat = TileMath.tileToLatLon(x: x, y: y, zoom: kRevealedTileParentZoom);
      final seLat = TileMath.tileToLatLon(x: x + 1, y: y + 1, zoom: kRevealedTileParentZoom);
      result.add(VisibleMirkTile(
        parentX: x, parentY: y,
        bitmap: bitmap,
        tileNorthLat: nwLat.lat, tileWestLon: nwLat.lon,
        tileSouthLat: seLat.lat, tileEastLon: seLat.lon,
      ));
    }
  }
  return result;
}
```

**IMPORTANT**: this reads ALL visible tiles on EVERY viewport change. For 50k tiles total in DB, at zoom 14 on a phone viewport, ~4-16 tiles are visible — well within budget. For 50k tiles over a sprawling session at zoom 10 (zoomed out), more tiles visible — the Phase 10 review will stress-test this with plan 09-08's fixture.

**`mapViewportProvider`** — may need to be introduced if not already there. Inspect `lib/application/providers/map_providers.dart` — Phase 07 likely has a `viewportCountryProvider` or similar. The mirk overlay needs the actual bbox, not just the country. If no `mapViewportProvider` exists, introduce it here or in a new `map_viewport_provider.dart` file. It reads `MapView.viewportUpdates` and publishes the latest `MirkViewportBbox`, debounced.

For Phase 09 MVP: a **simpler approach** is acceptable — have `MirkOverlay` itself subscribe to `MapView.viewportUpdates` and convert to `MirkViewportBbox` locally, then pass to the tile provider via `.family(bbox)`. Either way works; pick the one that fits Phase 07's provider pattern best.

**Debouncing**: 50 ms or leading-edge. The paint pass doesn't need every 60 Hz pan event; it reads the "latest" bbox at each paint.

class MirkOverlay extends ConsumerStatefulWidget {
  const MirkOverlay({super.key});
  @override
  ConsumerState<MirkOverlay> createState() => _MirkOverlayState();
}

class _MirkOverlayState extends ConsumerState<MirkOverlay>
    with SingleTickerProviderStateMixin {
  late final Ticker _ticker;
  Duration _frameElapsed = Duration.zero;

  @override
  void initState() {
    super.initState();
    _ticker = createTicker((elapsed) {
      _frameElapsed = elapsed;
      if (mounted) setState(() {});
    })..start();
  }

  @override
  void dispose() {
    _ticker.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final rendererAsync = ref.watch(activeMirkRendererProvider);
    final tilesAsync = ref.watch(visibleMirkTilesProvider);
    final viewport = ref.watch(mapViewportProvider);
    final sessionState = ref.watch(activeSessionControllerProvider);

    if (!rendererAsync.hasValue || !tilesAsync.hasValue || viewport == null) {
      return const SizedBox.shrink();
    }
    final renderer = rendererAsync.value!;
    final tiles = tilesAsync.value!;

    return RepaintBoundary(
      child: CustomPaint(
        painter: _MirkPainter(
          renderer: renderer,
          paintContext: MirkPaintContext(
            zoomLevel: viewport.zoomLevel,  // adjust per MapView surface
            pixelRatio: MediaQuery.of(context).devicePixelRatio,
            sessionElapsed: /* derived from session state */,
            viewportBbox: viewport,
            frameElapsed: _frameElapsed,
            currentFix: sessionState.valueOrNull?.whenOrNull(
              tracking: (_, fix) => fix,
            ),
            visibleTiles: tiles,
          ),
        ),
        size: Size.infinite,
      ),
    );
  }
}

class _MirkPainter extends CustomPainter {
  _MirkPainter({required this.renderer, required this.paintContext});
  final MirkRenderer renderer;
  final MirkPaintContext paintContext;

  @override
  void paint(Canvas canvas, Size size) => renderer.paint(canvas, size, paintContext);

  @override
  bool shouldRepaint(_MirkPainter old) => true;  // Ticker drives repaints
}
```

class MirkStylePickerSheet extends ConsumerWidget {
  const MirkStylePickerSheet({super.key, required this.sessionId});
  final SessionId sessionId;

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final builtinsAsync = ref.watch(builtinMirkStylesProvider);
    final sessionAsync = ref.watch(activeSessionControllerProvider);
    final currentStyleId = sessionAsync.valueOrNull?.whenOrNull(
      tracking: (s, _) => s.mirkStyleId,
    );

    return SafeArea(
      child: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          const Padding(
            padding: EdgeInsets.all(16),
            child: Text('Choisir un style de mirk', style: /* title */),
          ),
          builtinsAsync.when(
            loading: () => const Center(child: CircularProgressIndicator()),
            error: (_, __) => const Padding(
              padding: EdgeInsets.all(16),
              child: Text('Erreur de chargement des styles'),
            ),
            data: (styles) => Column(
              mainAxisSize: MainAxisSize.min,
              children: styles.map((style) {
                final isCurrent = style.id == currentStyleId;
                return ListTile(
                  title: Text(style.displayName),
                  trailing: isCurrent ? const Icon(Icons.check) : null,
                  onTap: () => _onSelect(context, ref, style),
                );
              }).toList(),
            ),
          ),
        ],
      ),
    );
  }

  Future<void> _onSelect(BuildContext context, WidgetRef ref, MirkStyle style) async {
    try {
      final ctrl = ref.read(mirkStyleSessionControllerProvider);
      await ctrl.select(sessionId: sessionId, styleId: style.id);
      if (!context.mounted) return;
      Navigator.of(context).pop();
    } on Exception catch (e) {
      if (!context.mounted) return;
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Erreur : \$e')),
      );
    }
  }
}
```
