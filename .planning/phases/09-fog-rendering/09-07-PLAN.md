import 'dart:async';

import 'package:riverpod_annotation/riverpod_annotation.dart';

import '../../domain/map/map_view.dart';
import '../../domain/mirk/mirk_viewport_bbox.dart';

/// Current MapLibre viewport bounds as a [MirkViewportBbox], or null
/// until the MapView is ready and the first viewport bounds settle.
///
/// Subscribes to [MapView.viewportUpdates] and republishes the bounds
/// derived from [MapView.queryViewportBounds] on each settled camera
/// change. Debounced 50 ms ŌĆö the overlay's Ticker reads `state` every
/// paint pass, so a coarse publish cadence suffices.
///
/// Phase 07 only exposes a scalar [MapViewportZoom] ŌĆö this provider is
/// the Phase 09 addition (plan 09-07 Task 1).
///
/// `keepAlive: true` ŌĆö bbox is a long-lived observable; tearing down the
/// subscription when the drawer closes would drop events during the gap.
@Riverpod(keepAlive: true)
class MapViewport extends _$MapViewport {
  Timer? _debounce;

  @override
  MirkViewportBbox? build() {
    final MapView? view = ref.watch(mapViewProvider);
    if (view == null) return null;
    final sub = view.viewportUpdates.listen(
      (_) => _scheduleRefresh(view),
      onError: (Object _, StackTrace _) { /* Phase 07 convention ŌĆö drop transient errors */ },
    );
    ref.onDispose(() {
      sub.cancel();
      _debounce?.cancel();
    });
    // Seed once on first build (mirror MapViewportZoom pattern).
    unawaited(_refreshNow(view));
    return null;
  }

  void _scheduleRefresh(MapView view) {
    _debounce?.cancel();
    _debounce = Timer(const Duration(milliseconds: 50), () => _refreshNow(view));
  }

  Future<void> _refreshNow(MapView view) async {
    try {
      final bbox = await view.queryViewportBounds();
      if (ref.mounted) state = bbox;
    } on Object {
      // queryViewportBounds can throw before MapLibre surface loads.
      // Benign ŌĆö next viewportUpdates emission retries.
    }
  }
}

/// Alias following the "xxxProvider" convention.
// ignore: non_constant_identifier_names
final mapViewportProvider = mapViewportProvider_;  // shim ŌĆö see generated .g.dart for actual name

**Implement `lib/application/providers/visible_mirk_tiles_provider.dart`** ŌĆö async `@Riverpod(keepAlive: false)` function:

```dart
@riverpod
Future<List<VisibleMirkTile>> visibleMirkTiles(Ref ref) async {
  final sessionState = ref.watch(activeSessionControllerProvider);
  final sessionId = sessionState.valueOrNull?.whenOrNull(
    tracking: (s, _) => s.id,
  );
  if (sessionId == null) return const [];

  final viewport = ref.watch(mapViewportProvider);  // from Task 1
  if (viewport == null) return const [];

  final nw = TileMath.latLonToTile(
      lat: viewport.north, lon: viewport.west, zoom: kRevealedTileParentZoom);
  final se = TileMath.latLonToTile(
      lat: viewport.south, lon: viewport.east, zoom: kRevealedTileParentZoom);

  final store = ref.watch(revealedTileStoreProvider);
  final result = <VisibleMirkTile>[];
  for (var y = nw.y; y <= se.y; y++) {
    for (var x = nw.x; x <= se.x; x++) {
      final row = await store.findByParent(
          sessionId: sessionId, parentX: x, parentY: y);
      // Null bitmap = no bits revealed yet for this tile. Still include
      // ŌĆö all-zero bitmap means the entire tile is fog.
      final bitmap = row?.bitmap ?? Uint8List(kRevealedTileBitmapBytes);
      final nwLatLon = TileMath.tileToLatLon(x: x, y: y, zoom: kRevealedTileParentZoom);
      final seLatLon = TileMath.tileToLatLon(x: x + 1, y: y + 1, zoom: kRevealedTileParentZoom);
      result.add(VisibleMirkTile(
        parentX: x, parentY: y,
        bitmap: bitmap,
        tileNorthLat: nwLatLon.lat, tileWestLon: nwLatLon.lon,
        tileSouthLat: seLatLon.lat, tileEastLon: seLatLon.lon,
      ));
    }
  }
  return result;
}
```

class MirkOverlay extends ConsumerStatefulWidget {
  const MirkOverlay({super.key});
  @override
  ConsumerState<MirkOverlay> createState() => _MirkOverlayState();
}

class _MirkOverlayState extends ConsumerState<MirkOverlay>
    with SingleTickerProviderStateMixin {
  late final Ticker _ticker;
  Duration _tickerElapsed = Duration.zero;

  @override
  void initState() {
    super.initState();
    _ticker = createTicker((elapsed) {
      _tickerElapsed = elapsed;
      if (mounted) setState(() {});
    })..start();
  }

  @override
  void dispose() {
    _ticker.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final rendererAsync = ref.watch(activeMirkRendererProvider);
    final tilesAsync = ref.watch(visibleMirkTilesProvider);
    final viewport = ref.watch(mapViewportProvider);
    final sessionState = ref.watch(activeSessionControllerProvider);
    final zoomState = ref.watch(mapViewportZoomProvider);  // Phase 07 existing

    if (!rendererAsync.hasValue || !tilesAsync.hasValue ||
        viewport == null || zoomState == null) {
      return const SizedBox.shrink();
    }
    final renderer = rendererAsync.value!;
    final tiles = tilesAsync.value!;
    final currentFix = sessionState.valueOrNull?.whenOrNull(
      tracking: (_, fix) => fix,
    );

    return RepaintBoundary(
      child: CustomPaint(
        painter: _MirkPainter(
          renderer: renderer,
          paintContext: MirkPaintContext(
            zoomLevel: zoomState,
            pixelRatio: MediaQuery.of(context).devicePixelRatio,
            sessionElapsed: _tickerElapsed,  // session-time proxy ŌĆö research consolidated on sessionElapsed
            viewportBbox: viewport,
            visibleTiles: tiles,
            currentFix: currentFix,
          ),
        ),
        size: Size.infinite,
      ),
    );
  }
}

class _MirkPainter extends CustomPainter {
  _MirkPainter({required this.renderer, required this.paintContext});
  final MirkRenderer renderer;
  final MirkPaintContext paintContext;
  @override
  void paint(Canvas canvas, Size size) => renderer.paint(canvas, size, paintContext);
  @override
  bool shouldRepaint(_MirkPainter old) => true;  // Ticker drives repaints
}

/// Fades the initial 20 m reveal from opacity 0 to 1 over
/// [kInitialRevealFadeInMs] (500 ms) at session start.
///
/// Uses a dedicated `AnimationController`, DECOUPLED from the main
/// [MirkOverlay] Ticker. The fade is a one-shot session-open animation
/// tied to [ActiveSessionController.startSession()] resolving (or the
/// first Fix arriving if no `lastKnownFix` was available). Coupling it
/// to the noise-tick frequency would make the fade duration depend on
/// the noise scale ŌĆö wrong abstraction.
///
/// Trigger:
/// * `activeSessionControllerProvider` state transitions from a non-tracking
///   state to `tracking(...)` with a non-null `lastFix` ŌåÆ forward play.
/// * Same transition but `lastFix == null` ŌåÆ waits on `lastFix` to become
///   non-null (via the next rebuild) then forward play.
/// * Session stops ŌåÆ controller resets to opacity 0 (no reverse fade in
///   Phase 09; polish candidate).
///
/// The wrapped [child] typically represents the session-open visual
/// (a circular ring indicator, a screen-wide alpha mask, etc.). Plan
/// 09-07 leaves the exact child to the [MapScreen] integration ŌĆö for
/// Phase 09 MVP the widget wraps a full-screen fog mask that fades in.
class MirkInitialRevealFade extends ConsumerStatefulWidget {
  const MirkInitialRevealFade({super.key, required this.child});
  final Widget child;
  @override
  ConsumerState<MirkInitialRevealFade> createState() => _MirkInitialRevealFadeState();
}

class _MirkInitialRevealFadeState extends ConsumerState<MirkInitialRevealFade>
    with SingleTickerProviderStateMixin {
  late final AnimationController _controller;
  bool _hasFadedIn = false;  // idempotence guard ŌĆö fade runs once per session

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: kInitialRevealFadeInMs),
    );
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final sessionState = ref.watch(activeSessionControllerProvider);
    final trackingFix = sessionState.valueOrNull?.whenOrNull(
      tracking: (_, fix) => fix,
    );
    if (trackingFix != null && !_hasFadedIn) {
      _hasFadedIn = true;
      _controller.forward(from: 0.0);
    } else if (sessionState.valueOrNull?.whenOrNull(tracking: (_, __) => true) == null) {
      // Session ended or not started ŌĆö reset for next session
      _controller.value = 0.0;
      _hasFadedIn = false;
    }

    return FadeTransition(
      opacity: CurvedAnimation(parent: _controller, curve: Curves.easeOut),
      child: widget.child,
    );
  }
}
